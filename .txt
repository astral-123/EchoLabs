-- ─────────────────────────────────────────────────────────────
-- UTILS
-- ─────────────────────────────────────────────────────────────
local function corner(parent, r)
    local c = Instance.new("UICorner", parent)
    c.CornerRadius = UDim.new(0, r or 4)
    return c
end
local function stroke(parent, color, thick, transp)
    local s = Instance.new("UIStroke", parent)
    s.Color = color or library.theme.surface
    s.Thickness = thick or 1
    s.Transparency = transp or 0
    return s
end
local function label(parent, text, size, font, color, xalign)
    local l = Instance.new("TextLabel", parent)
    l.Size = UDim2.new(1,0,1,0)
    l.BackgroundTransparency = 1
    l.Text = text or ""
    l.TextSize = size or 13
    l.Font = font or library.theme.font_reg
    l.TextColor3 = color or library.theme.text
    l.TextXAlignment = xalign or Enum.TextXAlignment.Left
    return l
end

function library:CreateWatermark()
    local sg = Instance.new("ScreenGui", coregui)
    sg.Name = "PhantomWatermark"

    f.Size = UDim2.fromOffset(195, 24)
    f.Position = UDim2.new(0, 12, 0, 12)

    f.BorderSizePixel = 0
    corner(f, 4)
    stroke(f, library.theme.surface, 1)

    local dot = Instance.new("Frame", f)
    dot.Size = UDim2.fromOffset(5,5)
    dot.Position = UDim2.new(0,7,0.5,-2)
    dot.BackgroundColor3 = library.theme.accent
    dot.BorderSizePixel = 0
    corner(dot, 10)

    local lbl = Instance.new("TextLabel", f)
    lbl.Size = UDim2.new(1,-18,1,0)
    lbl.Position = UDim2.new(0,16,0,0)
    lbl.BackgroundTransparency = 1
    lbl.Text = "phantom | 0 FPS | 0 MS"
    lbl.TextColor3 = library.theme.text
    lbl.Font = library.theme.font
    lbl.TextSize = 12
    lbl.TextXAlignment = Enum.TextXAlignment.Left

    run.RenderStepped:Connect(function()
        local fps = math.floor(1 / (run.RenderStepped:Wait()))
        local ok, ping = pcall(function()
            return math.floor(stats.Network.ServerStatsItem["Data Ping"]:GetValue())
        end)
        lbl.Text = "phantom | " .. fps .. " FPS | " .. (ok and ping or 0) .. " MS"
    end)
end

-- ─────────────────────────────────────────────────────────────
-- COLOR PICKER POPUP
-- ─────────────────────────────────────────────────────────────
local function createColorPicker(anchor_frame, current_color, on_change)
    -- Kill existing popup
    for _, c in ipairs(coregui:GetChildren()) do
        if c.Name == "PhantomColorPicker" then c:Destroy() end
    end

    local sg = Instance.new("ScreenGui", coregui)
    sg.Name = "PhantomColorPicker"
    sg.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

    local popup = Instance.new("Frame", sg)
    popup.Size = UDim2.fromOffset(210, 210)
    popup.BackgroundColor3 = library.theme.bg
    popup.BorderSizePixel = 0
    popup.ZIndex = 100
    corner(popup, 6)
    stroke(popup, library.theme.accent, 1, 0.5)

    -- Position near anchor
    local abs = anchor_frame.AbsolutePosition
    popup.Position = UDim2.fromOffset(abs.X, abs.Y + anchor_frame.AbsoluteSize.Y + 4)

    -- H S V current
    local h, s, v = Color3.toHSV(current_color)

    -- Preview box
    local preview = Instance.new("Frame", popup)
    preview.Size = UDim2.new(1, -16, 0, 22)
    preview.Position = UDim2.new(0, 8, 0, 8)
    preview.BackgroundColor3 = current_color
    preview.BorderSizePixel = 0
    corner(preview, 4)

    -- Hue gradient bar
    local hue_track = Instance.new("Frame", popup)
    hue_track.Size = UDim2.new(1,-16,0,10)
    hue_track.Position = UDim2.new(0,8,0,38)
    hue_track.BackgroundColor3 = Color3.new(1,1,1)
    hue_track.BorderSizePixel = 0
    corner(hue_track, 3)

    local hue_grad = Instance.new("UIGradient", hue_track)
    hue_grad.Color = ColorSequence.new({
        ColorSequenceKeypoint.new(0,    Color3.fromHSV(0,1,1)),
        ColorSequenceKeypoint.new(0.17, Color3.fromHSV(0.17,1,1)),
        ColorSequenceKeypoint.new(0.33, Color3.fromHSV(0.33,1,1)),
        ColorSequenceKeypoint.new(0.5,  Color3.fromHSV(0.5,1,1)),
        ColorSequenceKeypoint.new(0.67, Color3.fromHSV(0.67,1,1)),
        ColorSequenceKeypoint.new(0.83, Color3.fromHSV(0.83,1,1)),
        ColorSequenceKeypoint.new(1,    Color3.fromHSV(1,1,1)),
    })

    -- Hue thumb
    local hue_thumb = Instance.new("Frame", hue_track)
    hue_thumb.Size = UDim2.fromOffset(4, 14)
    hue_thumb.AnchorPoint = Vector2.new(0.5, 0.5)
    hue_thumb.Position = UDim2.new(h, 0, 0.5, 0)
    hue_thumb.BackgroundColor3 = Color3.new(1,1,1)
    hue_thumb.BorderSizePixel = 0
    corner(hue_thumb, 2)

    -- SV 2D picker (layered gradients)
    local sv_frame = Instance.new("Frame", popup)
    sv_frame.Size = UDim2.new(1,-16, 0, 100)
    sv_frame.Position = UDim2.new(0,8,0,56)
    sv_frame.BorderSizePixel = 0
    corner(sv_frame, 4)

    -- White-to-hue horizontal
    local sv_base = Instance.new("Frame", sv_frame)
    sv_base.Size = UDim2.new(1,0,1,0)
    sv_base.BackgroundColor3 = Color3.fromHSV(h,1,1)
    sv_base.BorderSizePixel = 0
    corner(sv_base, 4)

    local sv_sat_grad = Instance.new("UIGradient", sv_base)
    sv_sat_grad.Color = ColorSequence.new(Color3.new(1,1,1), Color3.fromHSV(h,1,1))

    -- Black overlay vertical
    local sv_dark = Instance.new("Frame", sv_base)
    sv_dark.Size = UDim2.new(1,0,1,0)
    sv_dark.BackgroundColor3 = Color3.new(0,0,0)
    sv_dark.BackgroundTransparency = 0
    sv_dark.BorderSizePixel = 0
    corner(sv_dark, 4)
    local sv_dark_grad = Instance.new("UIGradient", sv_dark)
    sv_dark_grad.Color = ColorSequence.new(Color3.new(0,0,0), Color3.new(0,0,0))
    sv_dark_grad.Transparency = NumberSequence.new({
        NumberSequenceKeypoint.new(0, 1),
        NumberSequenceKeypoint.new(1, 0),
    })
    sv_dark_grad.Rotation = 90

    -- SV thumb
    local sv_thumb = Instance.new("Frame", sv_frame)
    sv_thumb.Size = UDim2.fromOffset(8,8)
    sv_thumb.AnchorPoint = Vector2.new(0.5,0.5)
    sv_thumb.Position = UDim2.new(s, 0, 1-v, 0)
    sv_thumb.BackgroundColor3 = Color3.new(1,1,1)
    sv_thumb.BorderSizePixel = 0
    sv_thumb.ZIndex = 5
    corner(sv_thumb, 10)
    stroke(sv_thumb, Color3.new(0,0,0), 1)

    local function updateColor()
        local col = Color3.fromHSV(h, s, v)
        preview.BackgroundColor3 = col
        sv_base.BackgroundColor3 = Color3.fromHSV(h, 1, 1)
        sv_sat_grad.Color = ColorSequence.new(Color3.new(1,1,1), Color3.fromHSV(h,1,1))
        on_change(col)
    end

    -- Hue drag
    local hue_drag = false
    hue_track.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then hue_drag = true end
    end)
    uis.InputChanged:Connect(function(i)
        if hue_drag and i.UserInputType == Enum.UserInputType.MouseMovement then
            h = math.clamp((uis:GetMouseLocation().X - hue_track.AbsolutePosition.X) / hue_track.AbsoluteSize.X, 0, 1)
            hue_thumb.Position = UDim2.new(h, 0, 0.5, 0)
            updateColor()
        end
    end)
    uis.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then hue_drag = false end
    end)

    -- SV drag
    local sv_drag = false
    sv_frame.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then sv_drag = true end
    end)
    uis.InputChanged:Connect(function(i)
        if sv_drag and i.UserInputType == Enum.UserInputType.MouseMovement then
            local mx = uis:GetMouseLocation()
            s = math.clamp((mx.X - sv_frame.AbsolutePosition.X) / sv_frame.AbsoluteSize.X, 0, 1)
            v = 1 - math.clamp((mx.Y - sv_frame.AbsolutePosition.Y) / sv_frame.AbsoluteSize.Y, 0, 1)
            sv_thumb.Position = UDim2.new(s, 0, 1-v, 0)
            updateColor()
        end
    end)
    uis.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then sv_drag = false end
    end)

    -- Close button
    local close_btn = Instance.new("TextButton", popup)
    close_btn.Size = UDim2.new(1,-16,0,20)
    close_btn.Position = UDim2.new(0,8,1,-26)
    close_btn.BackgroundColor3 = library.theme.surface
    close_btn.Text = "Fermer"
    close_btn.TextColor3 = library.theme.dim
    close_btn.Font = library.theme.font
    close_btn.TextSize = 11
    close_btn.BorderSizePixel = 0
    corner(close_btn, 3)
    close_btn.MouseButton1Click:Connect(function() sg:Destroy() end)

    -- Close on click outside
    uis.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then
            local mx = uis:GetMouseLocation()
            local px, py = popup.AbsolutePosition.X, popup.AbsolutePosition.Y
            local pw, ph = popup.AbsoluteSize.X, popup.AbsoluteSize.Y
            if mx.X < px or mx.X > px+pw or mx.Y < py or mx.Y > py+ph then
                sg:Destroy()
            end
        end
    end)
end

-- ─────────────────────────────────────────────────────────────
-- WINDOW
-- ─────────────────────────────────────────────────────────────
function library:CreateWindow(title, hidekey)
    hidekey = hidekey or Enum.KeyCode.Insert

    local W = {
        tabs = {}, active_tab = nil,
        visible = true, dragging = false, is_interacting = false,
    }

    W.sg = Instance.new("ScreenGui", coregui)
    W.sg.Name = "PhantomUI"
    W.sg.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

    W.frame = Instance.new("Frame", W.sg)
    W.frame.Size = UDim2.fromOffset(500, 540)
    W.frame.Position = UDim2.new(1,-515,0.5,-270)
    W.frame.BackgroundColor3 = library.theme.bg
    W.frame.BorderSizePixel = 0
    W.frame.Active = true
    W.frame.ClipsDescendants = false   -- outer frame does NOT clip
    corner(W.frame, 6)
    stroke(W.frame, library.theme.surface, 1)

    -- ── Drag ──────────────────────────────────────────────────
    local dInput, dStart, dPos
    W.frame.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 and not W.is_interacting then
            W.dragging = true
            dStart = i.Position
            dPos = W.frame.Position
            i.Changed:Connect(function()
                if i.UserInputState == Enum.UserInputState.End then W.dragging = false end
            end)
        end
    end)
    W.frame.InputChanged:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseMovement then dInput = i end
    end)
    uis.InputChanged:Connect(function(i)
        if i == dInput and W.dragging then
            local d = i.Position - dStart
            W.frame.Position = UDim2.new(dPos.X.Scale, dPos.X.Offset+d.X, dPos.Y.Scale, dPos.Y.Offset+d.Y)
        end
    end)

    -- ── Header ────────────────────────────────────────────────
    local header = Instance.new("Frame", W.frame)
    header.Size = UDim2.new(1,0,0,32)
    header.BackgroundTransparency = 1

    local title_lbl = Instance.new("TextLabel", header)
    title_lbl.Size = UDim2.new(0.7,0,1,0)
    title_lbl.Position = UDim2.new(0,14,0,0)
    title_lbl.BackgroundTransparency = 1
    title_lbl.Text = "Phantom  |  Comfort  |  beta"
    title_lbl.TextColor3 = library.theme.accent
    title_lbl.Font = library.theme.font
    title_lbl.TextSize = 13
    title_lbl.TextXAlignment = Enum.TextXAlignment.Left

    local user_lbl = Instance.new("TextLabel", header)
    user_lbl.Size = UDim2.new(0.3,-12,1,0)
    user_lbl.Position = UDim2.new(0.7,0,0,0)
    user_lbl.BackgroundTransparency = 1
    user_lbl.Text = "user"
    user_lbl.TextColor3 = library.theme.dim
    user_lbl.Font = library.theme.font_reg
    user_lbl.TextSize = 12
    user_lbl.TextXAlignment = Enum.TextXAlignment.Right

    local sep1 = Instance.new("Frame", W.frame)
    sep1.Size = UDim2.new(1,0,0,1)
    sep1.Position = UDim2.new(0,0,0,32)
    sep1.BackgroundColor3 = library.theme.surface
    sep1.BorderSizePixel = 0

    -- ── Main tab bar ──────────────────────────────────────────
    local tabbar = Instance.new("Frame", W.frame)
    tabbar.Size = UDim2.new(1,0,0,28)
    tabbar.Position = UDim2.new(0,0,0,33)
    tabbar.BackgroundTransparency = 1
    tabbar.ClipsDescendants = false

    local tabbar_layout = Instance.new("UIListLayout", tabbar)
    tabbar_layout.FillDirection = Enum.FillDirection.Horizontal
    tabbar_layout.Padding = UDim.new(0,0)

    W.indicator = Instance.new("Frame", tabbar)
    W.indicator.Size = UDim2.fromOffset(60,2)
    W.indicator.Position = UDim2.new(0,0,1,-2)
    W.indicator.BackgroundColor3 = library.theme.accent
    W.indicator.BorderSizePixel = 0
    W.indicator.ZIndex = 10

    local sep2 = Instance.new("Frame", W.frame)
    sep2.Size = UDim2.new(1,0,0,1)
    sep2.Position = UDim2.new(0,0,0,61)
    sep2.BackgroundColor3 = library.theme.surface
    sep2.BorderSizePixel = 0

    -- ── Content (CLIPS to fix separator overflow) ─────────────
    W.content = Instance.new("Frame", W.frame)
    W.content.Size = UDim2.new(1,0,1,-63)
    W.content.Position = UDim2.new(0,0,0,63)
    W.content.BackgroundTransparency = 1
    W.content.ClipsDescendants = true   -- ← FIX separator overflow

    -- hide/show
    uis.InputBegan:Connect(function(i, gpe)
        if not gpe and i.KeyCode == hidekey then
            W.visible = not W.visible
            W.frame.Visible = W.visible
        end
    end)

                Position = UDim2.new(0, tab.btn.AbsolutePosition.X - tabbar.AbsolutePosition.X, 1,-2),

        if not W.active_tab then

        -- ============================================================
        -- makeScroll
        -- ============================================================
        local function makeScroll(parent)
            local sc = Instance.new("ScrollingFrame", parent)
            sc.Size = UDim2.new(1,0,1,0)
            sc.BackgroundTransparency = 1
            sc.ScrollBarThickness = 2
            sc.ScrollBarImageColor3 = library.theme.accent
            sc.CanvasSize = UDim2.new(0,0,0,0)
            local lay = Instance.new("UIListLayout", sc)
            lay.Padding = UDim.new(0,0)
            local pad = Instance.new("UIPadding", sc)
            pad.PaddingLeft  = UDim.new(0,10)
            pad.PaddingRight = UDim.new(0,8)
            pad.PaddingTop   = UDim.new(0,6)
            lay:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
                sc.CanvasSize = UDim2.new(0,0,0,lay.AbsoluteContentSize.Y+10)
            end)
            return sc
        end

        -- ============================================================
        -- buildElements
        -- ============================================================
        local function buildElements(scroll)
            local api = {}

            -- AddToggle
            function api:AddToggle(text, default, callback, kb_default)
                local row = Instance.new("TextButton", scroll)
                row.Size = UDim2.new(1,0,0,22)
                row.BackgroundTransparency = 1
                row.Text = ""
                row.BorderSizePixel = 0

                local box = Instance.new("Frame", row)
                box.Size = UDim2.fromOffset(13,13)
                box.Position = UDim2.new(0,0,0.5,-6)
                box.BackgroundColor3 = library.theme.surface
                box.BorderSizePixel = 0
                corner(box, 3)
                local bstroke = stroke(box, library.theme.dim, 1)

                local lbl_ = Instance.new("TextLabel", row)
                lbl_.Size = UDim2.new(0.65,0,1,0)
                lbl_.Position = UDim2.new(0,20,0,0)
                lbl_.BackgroundTransparency = 1
                lbl_.Text = text
                lbl_.TextColor3 = library.theme.dim
                lbl_.Font = library.theme.font_reg
                lbl_.TextSize = 13
                lbl_.TextXAlignment = Enum.TextXAlignment.Left

                if kb_default then
                    local kb_btn = Instance.new("TextButton", row)
                    kb_btn.Size = UDim2.fromOffset(60,16)
                    kb_btn.Position = UDim2.new(1,-62,0.5,-8)
                    kb_btn.BackgroundColor3 = library.theme.surface
                    kb_btn.Text = kb_default.Name
                    kb_btn.TextColor3 = library.theme.dim
                    kb_btn.Font = library.theme.font
                    kb_btn.TextSize = 11
                    kb_btn.BorderSizePixel = 0
                    corner(kb_btn, 3)
                    local listening = false
                    kb_btn.MouseButton1Click:Connect(function()
                        listening = true ; kb_btn.Text = "..."
                    end)
                    uis.InputBegan:Connect(function(i)
                        if listening and i.UserInputType == Enum.UserInputType.Keyboard then
                            listening = false ; kb_btn.Text = i.KeyCode.Name
                        end
                    end)
                end

                local val = {value = default}
                local function upd()
                    box.BackgroundColor3 = val.value and library.theme.accent or library.theme.surface
                    bstroke.Color        = val.value and library.theme.accent or library.theme.dim
                    lbl_.TextColor3      = val.value and library.theme.text or library.theme.dim
                    if callback then callback(val.value) end
                end
                row.MouseButton1Click:Connect(function()
                    val.value = not val.value ; upd()
                end)
                upd()
                return val
            end

            -- AddSlider
            function api:AddSlider(text, min, default, max, step, callback)
                local cont = Instance.new("Frame", scroll)
                cont.Size = UDim2.new(1,0,0,38)
                cont.BackgroundTransparency = 1

                local lbl_ = Instance.new("TextLabel", cont)
                lbl_.Size = UDim2.new(0.65,0,0,18)
                lbl_.BackgroundTransparency = 1

        -- ============================================================
        -- CreateSector
        -- ============================================================
        function tab:CreateSector(name, side)
            local bar     = side == "left"  and tab.left_bar     or tab.right_bar
            local content = side == "left"  and tab.left_content or tab.right_content
            local ind     = side == "left"  and tab.left_ind     or tab.right_ind
            local list    = side == "left"  and tab.left_subtabs or tab.right_subtabs

            local sector = {}

            sector.btn = Instance.new("TextButton", bar)
            sector.btn.Size = UDim2.fromOffset(0,26)
            sector.btn.AutomaticSize = Enum.AutomaticSize.X
            sector.btn.BackgroundTransparency = 1
            sector.btn.Text = name
            sector.btn.TextColor3 = library.theme.dim
            sector.btn.Font = library.theme.font
            sector.btn.TextSize = 12
            sector.btn.BorderSizePixel = 0

            sector.scroll = makeScroll(content)
            sector.scroll.Visible = false

            local api = buildElements(sector.scroll)
            for k,v in pairs(api) do sector[k] = v end

            sector.btn.MouseButton1Click:Connect(function()
                for _, s in ipairs(list) do
                    s.scroll.Visible = false
                    s.btn.TextColor3 = library.theme.dim
                end
                sector.scroll.Visible = true
                sector.btn.TextColor3 = library.theme.text
                tween:Create(ind, TweenInfo.new(0.12), {
                    Position = UDim2.new(0, sector.btn.AbsolutePosition.X - bar.AbsolutePosition.X, 1,-1),
                    Size     = UDim2.fromOffset(sector.btn.AbsoluteSize.X, 1),
                }):Play()
            end)

            if #list == 0 then
                sector.scroll.Visible = true
                sector.btn.TextColor3 = library.theme.text
                task.defer(function()
                    ind.Position = UDim2.new(0, sector.btn.AbsolutePosition.X - bar.AbsolutePosition.X, 1,-1)
                    ind.Size     = UDim2.fromOffset(sector.btn.AbsoluteSize.X, 1)
                end)
            end

            table.insert(list, sector)
            return sector
        end

        table.insert(W.tabs, tab)
        return tab
    end

    -- ================================================================
    -- Players Tab
    -- ================================================================
    function W:CreatePlayerTab()
        local pt      = self:CreateTab("Players")
        local list_s  = pt:CreateSector("Players", "left")
        local act_s   = pt:CreateSector("Actions", "right")
        local selected = nil

        local function refresh()
            for _, c in ipairs(list_s.scroll:GetChildren()) do
                if c:IsA("TextButton") then c:Destroy() end
            end
            for _, p in ipairs(players:GetPlayers()) do
                if p ~= players.LocalPlayer then
                    local pb = Instance.new("TextButton", list_s.scroll)
                    pb.Size = UDim2.new(1,0,0,26)
                    pb.BackgroundColor3 = library.theme.surface
                    pb.Text = "  "..p.Name
                    pb.TextColor3 = library.theme.dim
                    pb.Font = library.theme.font_reg
                    pb.TextSize = 12
                    pb.TextXAlignment = Enum.TextXAlignment.Left
                    pb.BorderSizePixel = 0
                    corner(pb, 4)
                    pb.MouseButton1Click:Connect(function()
                        selected = p
                        for _, c2 in ipairs(list_s.scroll:GetChildren()) do
                            if c2:IsA("TextButton") then
                                c2.BackgroundColor3 = library.theme.surface
                                c2.TextColor3 = library.theme.dim
                            end
                        end
                        pb.BackgroundColor3 = library.theme.accent:Lerp(Color3.new(0,0,0),0.5)
                        pb.TextColor3 = library.theme.text
                    end)
                end
            end
        end

        act_s:AddButton("Teleport to Player", function()
            if selected and selected.Character then
                players.LocalPlayer.Character:MoveTo(selected.Character.HumanoidRootPart.Position)
            end
        end)
        act_s:AddButton("Spectate", function()
            if selected and selected.Character then
                workspace.CurrentCamera.CameraSubject = selected.Character.Humanoid
            end
        end)
        act_s:AddButton("Stop Spectate", function()
            workspace.CurrentCamera.CameraSubject = players.LocalPlayer.Character.Humanoid
        end)

        refresh()
        players.PlayerAdded:Connect(refresh)
        players.PlayerRemoving:Connect(refresh)
    end

    -- ================================================================
    -- Options Tab  (couleurs + FPS Booster)
    -- ================================================================
    function W:CreateOptionsTab()
        local ot    = self:CreateTab("Options")
        local col_s = ot:CreateSector("Couleurs", "left")
        local cfg_s = ot:CreateSector("Config",   "right")

        -- ── Color pickers ─────────────────────────────────────
        col_s:AddColorPicker("Accent", library.theme.accent, function(c)
            library.theme.accent = c
            -- Update all accent-colored elements live would require refs,
            -- simplest: user sees change on next UI element interaction.
            -- For the indicator we can update directly:
            W.indicator.BackgroundColor3 = c
        end)

        col_s:AddColorPicker("Texte principal", library.theme.text, function(c)
            library.theme.text = c
        end)

        col_s:AddColorPicker("Texte secondaire", library.theme.dim, function(c)
            library.theme.dim = c
        end)

        col_s:AddColorPicker("Fond", library.theme.bg, function(c)
            library.theme.bg = c
            W.frame.BackgroundColor3 = c
        end)

        col_s:AddColorPicker("Surface", library.theme.surface, function(c)
            library.theme.surface = c
        end)

        -- Quick presets
        col_s:AddLabel("Presets")
        col_s:AddButton("Rose (défaut)", function()
            library.theme.accent = Color3.fromRGB(255,0,140)
            W.indicator.BackgroundColor3 = library.theme.accent
        end)
        col_s:AddButton("Violet", function()
            library.theme.accent = Color3.fromRGB(120,60,255)
            W.indicator.BackgroundColor3 = library.theme.accent
        end)
        col_s:AddButton("Cyan", function()
            library.theme.accent = Color3.fromRGB(0,200,255)
            W.indicator.BackgroundColor3 = library.theme.accent
        end)

        -- ── Config / keybind / FPS ─────────────────────────────
        cfg_s:AddKeybind("Toggle UI", hidekey, function(k) hidekey = k end)

        cfg_s:AddToggle("FPS Booster", false, function(state)
            if state then
                settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
                workspace.StreamingEnabled = false
                pcall(function()
                    game:GetService("UserSettings"):SetService("RenderSettings").SavedQualityLevel = Enum.QualityLevel.Level01
                end)
            else
                settings().Rendering.QualityLevel = Enum.QualityLevel.Automatic
            end
        end)

        cfg_s:AddToggle("Anti-Lag", false, function(state)
            if state then
                lighting.GlobalShadows = false
                lighting.FogEnd = 1e6
                for _, v in ipairs(lighting:GetChildren()) do
                    if v:IsA("BlurEffect") or v:IsA("DepthOfFieldEffect") or
                       v:IsA("SunRaysEffect") or v:IsA("ColorCorrectionEffect") or
                       v:IsA("BloomEffect") then
                        v.Enabled = false
                    end
                end
            else
                lighting.GlobalShadows = true
            end
        end)
    end

    return W
end

function library:Notify(title, text, duration)
    local sg = Instance.new("ScreenGui", coregui)
    sg.Name = "PhantomNotif"
    local f = Instance.new("Frame", sg)
    f.Size = UDim2.fromOffset(210,46)
    f.Position = UDim2.new(1,10,1,-62)
    f.BackgroundColor3 = library.theme.bg
    f.BorderSizePixel = 0
    corner(f,5)
    stroke(f, library.theme.accent, 1, 0.4)

    local t_ = Instance.new("TextLabel",f)
    t_.Size=UDim2.new(1,-8,0,18);t_.Position=UDim2.new(0,8,0,5)
    t_.BackgroundTransparency=1;t_.Text=title;t_.TextColor3=library.theme.accent
    t_.Font=library.theme.font;t_.TextSize=12;t_.TextXAlignment=Enum.TextXAlignment.Left

    local b_ = Instance.new("TextLabel",f)
    b_.Size=UDim2.new(1,-8,0,16);b_.Position=UDim2.new(0,8,0,24)
    b_.BackgroundTransparency=1;b_.Text=text;b_.TextColor3=library.theme.dim
    b_.Font=library.theme.font_reg;b_.TextSize=11;b_.TextXAlignment=Enum.TextXAlignment.Left

    tween:Create(f,TweenInfo.new(0.2,Enum.EasingStyle.Back),{Position=UDim2.new(1,-222,1,-62)}):Play()
    task.delay(duration or 5, function()
        tween:Create(f,TweenInfo.new(0.15),{Position=UDim2.new(1,10,1,-62)}):Play()
        task.delay(0.2,function() sg:Destroy() end)
    end)
end

return library
