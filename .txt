local CF = "phantom_configs"
local AL = "phantom_autoload.txt"

local function safe(fn, ...) local ok, r = pcall(fn, ...); return ok and r or nil end

local function ensureFolder()
    if isfolder and not isfolder(CF) and makefolder then makefolder(CF) end
end
local function listConfigs()
    ensureFolder()
    local files = safe(listfiles, CF) or {}
    local names = {}
    for _, p in ipairs(files) do
        local n = p:match("[/\\ ]?([^/\\ ]+)%.json$")
        if n then table.insert(names, n) end
    end
    return names
end
local function saveConfigFile(name, data)
    ensureFolder()
    if writefile then writefile(CF.."/"..name..".json", http:JSONEncode(data)); return true end
    return false
end
local function loadConfigFile(name)
    local raw = safe(readfile, CF.."/"..name..".json")
    if raw then return safe(http.JSONDecode, http, raw) end
    return nil
end
local function deleteConfigFile(name)
    local p = CF.."/"..name..".json"
    if isfile and isfile(p) and delfile then delfile(p) end
end
local function getAutoLoad()
    if isfile and isfile(AL) and readfile then
        local v = safe(readfile, AL)
        return (v and v ~= "") and v or nil
    end
end
local function setAutoLoad(name)
    if writefile then writefile(AL, name or "") end
end

local function corner(p, r)
    local c = Instance.new("UICorner", p)
    c.CornerRadius = UDim.new(0, r or 0) -- Rendu plus rectangulaire pour BO style
    return c
end
local function stroke(p, col, thick, transp)
    local s = Instance.new("UIStroke", p)
    s.Color       = col   or library.theme.border
    s.Thickness   = thick or 1
    s.Transparency = transp or 0
    s.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
    return s
end
local function sep(parent, ypos, xoff, xsize)
    local s = Instance.new("Frame", parent)
    s.Size            = UDim2.new(1, xsize or 0, 0, 1)
    s.Position        = UDim2.new(0, xoff or 0, 0, ypos)
    s.BackgroundColor3 = library.theme.border
    s.BorderSizePixel = 0
    return s
end

local function mkColorPicker(anchor, cur_col, on_change)
    for _, c in ipairs(coregui:GetChildren()) do
        if c.Name == "PhCP" then c:Destroy() end
    end
    local sg  = Instance.new("ScreenGui", coregui); sg.Name = "PhCP"
    sg.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

    local pop = Instance.new("Frame", sg)
    pop.Size            = UDim2.fromOffset(212, 218)
    pop.BackgroundColor3 = library.theme.bg
    pop.BorderSizePixel = 0; pop.ZIndex = 100
    stroke(pop, library.theme.border, 1)
    local abs = anchor.AbsolutePosition
    pop.Position = UDim2.fromOffset(abs.X, abs.Y + anchor.AbsoluteSize.Y + 4)

    local h, s, v = Color3.toHSV(cur_col)

    -- Preview
    local prev = Instance.new("Frame", pop)
    prev.Size = UDim2.new(1, -16, 0, 22); prev.Position = UDim2.new(0, 8, 0, 8)
    prev.BackgroundColor3 = cur_col; prev.BorderSizePixel = 0
    stroke(prev, library.theme.border, 1)

    -- Hue bar
    local ht = Instance.new("Frame", pop)
    ht.Size = UDim2.new(1, -16, 0, 10); ht.Position = UDim2.new(0, 8, 0, 38)
    ht.BackgroundColor3 = Color3.new(1,1,1); ht.BorderSizePixel = 0
    local hg = Instance.new("UIGradient", ht)
    hg.Color = ColorSequence.new({
        ColorSequenceKeypoint.new(0,    Color3.fromHSV(0,    1, 1)),
        ColorSequenceKeypoint.new(0.17, Color3.fromHSV(0.17, 1, 1)),
        ColorSequenceKeypoint.new(0.33, Color3.fromHSV(0.33, 1, 1)),
        ColorSequenceKeypoint.new(0.5,  Color3.fromHSV(0.5,  1, 1)),
        ColorSequenceKeypoint.new(0.67, Color3.fromHSV(0.67, 1, 1)),
        ColorSequenceKeypoint.new(0.83, Color3.fromHSV(0.83, 1, 1)),
        ColorSequenceKeypoint.new(1,    Color3.fromHSV(1,    1, 1)),
    })
    local hth = Instance.new("Frame", ht)
    hth.Size = UDim2.fromOffset(4, 14); hth.AnchorPoint = Vector2.new(0.5, 0.5)
    hth.Position = UDim2.new(h, 0, 0.5, 0); hth.BackgroundColor3 = Color3.new(1,1,1)
    hth.BorderSizePixel = 0; stroke(hth, Color3.new(0,0,0), 1)

    -- SV box
    local svf = Instance.new("Frame", pop)
    svf.Size = UDim2.new(1, -16, 0, 108); svf.Position = UDim2.new(0, 8, 0, 56)
    svf.BorderSizePixel = 0
    local svb = Instance.new("Frame", svf)
    svb.Size = UDim2.new(1, 0, 1, 0); svb.BackgroundColor3 = Color3.fromHSV(h, 1, 1)
    svb.BorderSizePixel = 0
    local sg2 = Instance.new("UIGradient", svb)
    sg2.Color = ColorSequence.new(Color3.new(1,1,1), Color3.fromHSV(h, 1, 1))
    local svdk = Instance.new("Frame", svb)
    svdk.Size = UDim2.new(1,0,1,0); svdk.BackgroundColor3 = Color3.new(0,0,0)
    svdk.BorderSizePixel = 0
    local dg = Instance.new("UIGradient", svdk)
    dg.Color = ColorSequence.new(Color3.new(0,0,0), Color3.new(0,0,0))
    dg.Transparency = NumberSequence.new({NumberSequenceKeypoint.new(0,1), NumberSequenceKeypoint.new(1,0)})
    dg.Rotation = 90
    local svth = Instance.new("Frame", svf)
    svth.Size = UDim2.fromOffset(9, 9); svth.AnchorPoint = Vector2.new(0.5, 0.5)
    svth.Position = UDim2.new(s, 0, 1-v, 0); svth.BackgroundColor3 = Color3.new(1,1,1)
    svth.BorderSizePixel = 0; svth.ZIndex = 5; corner(svth, 10)
    stroke(svth, Color3.new(0,0,0), 1)

    local function upd()
        local col = Color3.fromHSV(h, s, v)
        prev.BackgroundColor3 = col
        svb.BackgroundColor3 = Color3.fromHSV(h, 1, 1)
        sg2.Color = ColorSequence.new(Color3.new(1,1,1), Color3.fromHSV(h, 1, 1))
        on_change(col)
    end

    local hd, svd = false, false
    ht.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then hd = true end end)
    svf.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then svd = true end end)
    uis.InputChanged:Connect(function(i)
        if i.UserInputType ~= Enum.UserInputType.MouseMovement then return end
        local mx = uis:GetMouseLocation()
        if hd then
            h = math.clamp((mx.X - ht.AbsolutePosition.X) / ht.AbsoluteSize.X, 0, 1)
            hth.Position = UDim2.new(h, 0, 0.5, 0); upd()
        end
        if svd then
            s = math.clamp((mx.X - svf.AbsolutePosition.X) / svf.AbsoluteSize.X, 0, 1)
            v = 1 - math.clamp((mx.Y - svf.AbsolutePosition.Y) / svf.AbsoluteSize.Y, 0, 1)
            svth.Position = UDim2.new(s, 0, 1-v, 0); upd()
        end
    end)
    uis.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then hd = false; svd = false end
    end)

    local cb = Instance.new("TextButton", pop)
    cb.Size = UDim2.new(1, -16, 0, 18); cb.Position = UDim2.new(0, 8, 1, -24)
    cb.BackgroundColor3 = library.theme.surface; cb.Text = "Fermer"
    cb.TextColor3 = library.theme.dim; cb.Font = library.theme.font; cb.TextSize = 11
    cb.BorderSizePixel = 0
    stroke(cb, library.theme.border, 1)
    cb.MouseButton1Click:Connect(function() sg:Destroy() end)

    local conn
    conn = uis.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then
            local mx = uis:GetMouseLocation()
            local px, py = pop.AbsolutePosition.X, pop.AbsolutePosition.Y
            if mx.X < px or mx.X > px+pop.AbsoluteSize.X or mx.Y < py or mx.Y > py+pop.AbsoluteSize.Y then
                sg:Destroy(); conn:Disconnect()
            end
        end
    end)
end

function library:CreateWatermark()
    local sg = Instance.new("ScreenGui", coregui); sg.Name = "PhantomWM"
    local f  = Instance.new("Frame", sg)
    f.Size = UDim2.fromOffset(200, 24); f.Position = UDim2.new(0, 10, 0, 10)
    f.BackgroundColor3 = library.theme.bg; f.BorderSizePixel = 0
    stroke(f, library.theme.border, 1)
    local dot = Instance.new("Frame", f)
    dot.Size = UDim2.fromOffset(5, 5); dot.Position = UDim2.new(0, 7, 0.5, -2)
    dot.BackgroundColor3 = library.theme.accent; dot.BorderSizePixel = 0; corner(dot, 10)
    local lbl = Instance.new("TextLabel", f)
    lbl.Size = UDim2.new(1, -18, 1, 0); lbl.Position = UDim2.new(0, 16, 0, 0)
    lbl.BackgroundTransparency = 1; lbl.TextColor3 = library.theme.text
    lbl.Font = library.theme.font; lbl.TextSize = 12
    lbl.TextXAlignment = Enum.TextXAlignment.Left
    task.spawn(function()
        local lastT = os.clock()
        while sg and sg.Parent do
            run.RenderStepped:Wait()
            local now = os.clock(); local dt = now - lastT; lastT = now
            local fps = dt > 0 and math.floor(1/dt) or 0
            local ping = "0"
            pcall(function()
                ping = tostring(math.floor(stats.Network.ServerStatsItem["Data Ping"]:GetValue()))
            end)
            lbl.Text = "phantom | "..fps.." FPS | "..ping.." MS"
        end
    end)
end

    end
    local f = Instance.new("Frame", sg)

    f.BackgroundColor3 = library.theme.bg; f.BorderSizePixel = 0

    t.Size = UDim2.new(1, -10, 0, 20); t.Position = UDim2.fromOffset(10, 5)
    t.BackgroundTransparency = 1; t.Text = title; t.TextColor3 = library.theme.accent
    t.Font = library.theme.font; t.TextSize = 13; t.TextXAlignment = Enum.TextXAlignment.Left
    local m = Instance.new("TextLabel", f)
    m.Size = UDim2.new(1, -10, 0, 20); m.Position = UDim2.fromOffset(10, 22)
    m.BackgroundTransparency = 1; m.Text = text; m.TextColor3 = library.theme.text
    m.Font = library.theme.font_reg; m.TextSize = 12; m.TextXAlignment = Enum.TextXAlignment.Left

    task.delay(dur or 3, function()
        tw:Create(f, TweenInfo.new(0.3), { Position = UDim2.new(1, 20, 1, -60) }):Play()
        task.wait(0.3); f:Destroy()
    end)

            arrow.TextColor3       = library.theme.accent
            arrow.Font             = library.theme.font
            arrow.TextSize         = 9
            arrow.TextXAlignment   = Enum.TextXAlignment.Center

            local list = Instance.new("Frame", W.sg)
            list.Size             = UDim2.fromOffset(0, 0)
            list.BackgroundColor3 = library.theme.surface
            list.Visible          = false
            list.ZIndex           = 200
            list.BorderSizePixel  = 0
            list.ClipsDescendants = true
            stroke(list, library.theme.border, 1)
            Instance.new("UIListLayout", list)

            local open = false
            local function closeList()
                open = false; list.Visible = false
                tw:Create(list,  TweenInfo.new(0.08), { Size   = UDim2.fromOffset(drop.AbsoluteSize.X, 0) }):Play()
                tw:Create(arrow, TweenInfo.new(0.1),  { Rotation = 0 }):Play()
            end
            local function openList()
                local abs = drop.AbsolutePosition
                local w   = drop.AbsoluteSize.X
                local h   = #options * 20
                list.Position = UDim2.fromOffset(abs.X, abs.Y + drop.AbsoluteSize.Y + 2)
                list.Size     = UDim2.fromOffset(w, 0)
                list.Visible  = true
                tw:Create(list,  TweenInfo.new(0.1), { Size   = UDim2.fromOffset(w, h) }):Play()
                tw:Create(arrow, TweenInfo.new(0.1), { Rotation = 180 }):Play()
            end

            drop.MouseButton1Click:Connect(function()
                if open then closeList() else openList() end
                open = not open
            end)

            local function selOpt(opt)
                drop.Text   = "  "..opt; state.value = opt
                if flagName then library.flags[flagName] = opt end
                closeList(); open = false
                if callback then callback(opt) end
            end
            state.setValue = function(opt) drop.Text = "  "..opt; state.value = opt; if callback then callback(opt) end end

            for _, opt in ipairs(options) do
                local ob = Instance.new("TextButton", list)
                ob.Size             = UDim2.new(1, 0, 0, 20)
                ob.BackgroundTransparency = 1
                ob.Text             = "  "..opt
                ob.TextColor3       = library.theme.dim
                ob.Font             = library.theme.font_reg
                ob.TextSize         = 12
                ob.TextXAlignment   = Enum.TextXAlignment.Left
                ob.ZIndex           = 201
                ob.MouseEnter:Connect(function() ob.TextColor3 = library.theme.text end)
                ob.MouseLeave:Connect(function() ob.TextColor3 = library.theme.dim end)
                ob.MouseButton1Click:Connect(function() selOpt(opt) end)
            end

            uis.InputBegan:Connect(function(i)
                if open and i.UserInputType == Enum.UserInputType.MouseButton1 then
                    local mx = uis:GetMouseLocation()
                    local function inBox(f)
                        return mx.X >= f.AbsolutePosition.X and mx.X <= f.AbsolutePosition.X + f.AbsoluteSize.X
                           and mx.Y >= f.AbsolutePosition.Y and mx.Y <= f.AbsolutePosition.Y + f.AbsoluteSize.Y
                    end
                    if not inBox(list) and not inBox(drop) then closeList(); open = false end
                end
            end)
            return state
        end

    function W:CreateTab(name)
        local tab = { name = name, sectors = {} }

        tab.btn = Instance.new("TextButton", tabbar)
        tab.btn.Size               = UDim2.fromOffset(100, 40)
        tab.btn.BackgroundTransparency = 1
        tab.btn.Text               = name
        tab.btn.TextColor3         = library.theme.dim
        tab.btn.Font               = library.theme.font
        tab.btn.TextSize           = 13
        tab.btn.BorderSizePixel    = 0

        tab.frame = Instance.new("Frame", W.content)
        tab.frame.Size             = UDim2.new(1, 0, 1, 0)
        tab.frame.BackgroundTransparency = 1
        tab.frame.Visible          = false
        
        local gl = Instance.new("UIGridLayout", tab.frame)
        gl.CellPadding = UDim2.fromOffset(10, 10)
        gl.CellSize = UDim2.new(0.5, -5, 1, 0)

        tab.btn.MouseButton1Click:Connect(function()
            for _, t in ipairs(W.tabs) do
                t.frame.Visible    = false
                t.btn.TextColor3   = library.theme.dim
            end
            tab.frame.Visible  = true
            tab.btn.TextColor3 = library.theme.text
            tw:Create(W.indicator, TweenInfo.new(0.12), {
                Size = UDim2.new(0, tab.btn.AbsoluteSize.X, 0, 2),
                Position = UDim2.new(0, tab.btn.AbsolutePosition.X - tabbar.AbsolutePosition.X, 0, 0)
            }):Play()
        end)

        function tab:CreateSector(sname)
            local s = W:CreateSector(sname, nil, tab.frame)
            table.insert(tab.sectors, s)
            return s
        end

        table.insert(W.tabs, tab)
        if #W.tabs == 1 then
            tab.frame.Visible = true
            tab.btn.TextColor3 = library.theme.text
            task.spawn(function()
                task.wait(0.1)
                W.indicator.Size = UDim2.new(0, tab.btn.AbsoluteSize.X, 0, 2)
                W.indicator.Position = UDim2.new(0, tab.btn.AbsolutePosition.X - tabbar.AbsolutePosition.X, 0, 0)
            end)
        end
        return tab
    end

    return W
