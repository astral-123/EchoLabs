        text_dark   = Color3.fromRGB(130, 130, 150),

local uis         = game:GetService("UserInputService")

-- ─── Helpers ──────────────────────────────────────────────────────────────────

local function sc(n) return math.round(n * library.scale) end  -- scale integer

-- ─── Watermark ────────────────────────────────────────────────────────────────

function library:CreateWatermark()
    local wm = Instance.new("ScreenGui", coregui)
    wm.Name  = "EclipseWatermark"

    local frame = Instance.new("Frame", wm)
    frame.Size             = UDim2.fromOffset(210, 26)
    frame.Position         = UDim2.new(0, 14, 0, 14)
    frame.BackgroundColor3 = library.theme.background
    frame.BorderSizePixel  = 0
    Instance.new("UICorner", frame).CornerRadius = UDim.new(0, 5)

    local stroke = Instance.new("UIStroke", frame)
    stroke.Color     = library.theme.accent
    stroke.Thickness = 1
    stroke.Transparency = 0.5

    local label = Instance.new("TextLabel", frame)
    label.Size               = UDim2.new(1, -10, 1, 0)
    label.Position           = UDim2.new(0, 8, 0, 0)
    label.BackgroundTransparency = 1
    label.Text               = "Eclipse  |  0 FPS  |  0 MS"
    label.TextColor3         = library.theme.text
    label.Font               = library.theme.font
    label.TextSize           = 12
    label.TextXAlignment     = Enum.TextXAlignment.Left

    -- Accent dot
    local dot = Instance.new("Frame", frame)
    dot.Size             = UDim2.fromOffset(5, 5)
    dot.Position         = UDim2.new(1, -10, 0.5, -2)
    dot.BackgroundColor3 = library.theme.accent
    dot.BorderSizePixel  = 0
    Instance.new("UICorner", dot).CornerRadius = UDim.new(1, 0)

    runservice.RenderStepped:Connect(function()
        local ok, fps = pcall(function()
            return math.floor(1 / runservice.RenderStepped:Wait())
        end)
        local ok2, ping = pcall(function()
            return math.floor(stats.Network.ServerStatsItem["Data Ping"]:GetValue())
        end)
        label.Text = "Eclipse  |  " .. (ok and fps or "?") .. " FPS  |  " .. (ok2 and ping or "?") .. " MS"
    end)
end

-- ─── CreateWindow ─────────────────────────────────────────────────────────────

function library:CreateWindow(name, size, hidebutton)
    local W = {
        tabs           = {},
        active_tab     = nil,
        hidebutton     = hidebutton or Enum.KeyCode.Insert,
        visible        = true,
        dragging       = false,
        is_interacting = false,
        sector_index   = 0,
    }

    local BASE_W = size and size.X or 680
    local BASE_H = size and size.Y or 520

    W.main        = Instance.new("ScreenGui", coregui)
    W.main.Name   = name

    W.frame                  = Instance.new("Frame", W.main)
    W.frame.Size             = UDim2.fromOffset(sc(BASE_W), sc(BASE_H))
    W.frame.Position         = UDim2.new(0.62, 0, 0.5, -sc(BASE_H / 2))
    W.frame.BackgroundColor3 = library.theme.background
    W.frame.BorderSizePixel  = 0
    W.frame.Active           = true
    W.frame.ClipsDescendants = false

    local corner = Instance.new("UICorner", W.frame)
    corner.CornerRadius = UDim.new(0, 8)

    -- Outer glow stroke
    local stroke = Instance.new("UIStroke", W.frame)
    stroke.Color       = library.theme.accent
    stroke.Thickness   = 1
    stroke.Transparency = 0.6

    -- ── Drag ──────────────────────────────────────────────────────────────────
    local dragInput, dragStart, startPos
    W.frame.InputBegan:Connect(function(inp)
        if (inp.UserInputType == Enum.UserInputType.MouseButton1) and not W.is_interacting then
            W.dragging = true
            dragStart  = inp.Position
            startPos   = W.frame.Position
            inp.Changed:Connect(function()
                if inp.UserInputState == Enum.UserInputState.End then W.dragging = false end
            end)
        end
    end)
    W.frame.InputChanged:Connect(function(inp)
        if inp.UserInputType == Enum.UserInputType.MouseMovement then dragInput = inp end
    end)
    uis.InputChanged:Connect(function(inp)
        if inp == dragInput and W.dragging then
            local d = inp.Position - dragStart
            W.frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + d.X,
                                          startPos.Y.Scale, startPos.Y.Offset + d.Y)
        end
    end)

    -- ── Header ────────────────────────────────────────────────────────────────
    local header = Instance.new("Frame", W.frame)
    header.Size                  = UDim2.new(1, -30, 0, 40)
    header.Position              = UDim2.new(0, 15, 0, 8)
    header.BackgroundTransparency = 1

    -- Logo bar accent
    local logo_bar = Instance.new("Frame", header)
    logo_bar.Size             = UDim2.fromOffset(3, 20)
    logo_bar.Position         = UDim2.new(0, 0, 0.5, -10)
    logo_bar.BackgroundColor3 = library.theme.accent
    logo_bar.BorderSizePixel  = 0
    Instance.new("UICorner", logo_bar).CornerRadius = UDim.new(0, 2)

    local title = Instance.new("TextLabel", header)
    title.Size               = UDim2.new(1, -10, 1, 0)
    title.Position           = UDim2.new(0, 10, 0, 0)
    title.Text               = "Eclipse  •  Comfort  •  beta"
    title.TextColor3         = library.theme.text
    title.Font               = library.theme.font
    title.TextSize           = 15
    title.TextXAlignment     = Enum.TextXAlignment.Left
    title.BackgroundTransparency = 1

    -- Watermark-style info top-right
    local info = Instance.new("TextLabel", header)
    info.Size               = UDim2.new(0.4, 0, 1, 0)
    info.Position           = UDim2.new(0.6, 0, 0, 0)
    info.Text               = "zixlr"
    info.TextColor3         = library.theme.text_dark
    info.Font               = library.theme.font
    info.TextSize           = 12
    info.TextXAlignment     = Enum.TextXAlignment.Right
    info.BackgroundTransparency = 1

    -- Separator line
    local sep = Instance.new("Frame", W.frame)
    sep.Size             = UDim2.new(1, -30, 0, 1)
    sep.Position         = UDim2.new(0, 15, 0, 50)
    sep.BackgroundColor3 = library.theme.foreground
    sep.BorderSizePixel  = 0

    -- ── Tab container (horizontal) ────────────────────────────────────────────
    W.tab_container          = Instance.new("Frame", W.frame)
    W.tab_container.Size     = UDim2.new(1, -30, 0, 28)
    W.tab_container.Position = UDim2.new(0, 15, 0, 56)
    W.tab_container.BackgroundTransparency = 1

    local tab_layout = Instance.new("UIListLayout", W.tab_container)
    tab_layout.FillDirection = Enum.FillDirection.Horizontal
    tab_layout.Padding       = UDim.new(0, 18)

    -- Tab indicator (underline)
    W.tab_indicator          = Instance.new("Frame", W.tab_container)
    W.tab_indicator.Size     = UDim2.fromOffset(50, 2)
    W.tab_indicator.Position = UDim2.new(0, 0, 1, 0)
    W.tab_indicator.BackgroundColor3 = library.theme.accent
    W.tab_indicator.BorderSizePixel  = 0
    W.tab_indicator.ZIndex  = 5
    Instance.new("UICorner", W.tab_indicator).CornerRadius = UDim.new(0, 1)

    -- Separator 2
    local sep2 = Instance.new("Frame", W.frame)
    sep2.Size             = UDim2.new(1, -30, 0, 1)
    sep2.Position         = UDim2.new(0, 15, 0, 86)
    sep2.BackgroundColor3 = library.theme.foreground
    sep2.BorderSizePixel  = 0

    -- ── Content ───────────────────────────────────────────────────────────────
    W.content          = Instance.new("Frame", W.frame)
    W.content.Size     = UDim2.new(1, -30, 1, -100)
    W.content.Position = UDim2.new(0, 15, 0, 94)
    W.content.BackgroundTransparency = 1

    -- ── Hide toggle ───────────────────────────────────────────────────────────
    uis.InputBegan:Connect(function(inp, gpe)
        if not gpe and inp.KeyCode == W.hidebutton then
            W.visible    = not W.visible
            W.frame.Visible = W.visible
        end
    end)

    -- ── Function: rescale window ──────────────────────────────────────────────
    function W:SetScale(factor)
        library.scale    = factor
        W.frame.Size     = UDim2.fromOffset(sc(BASE_W), sc(BASE_H))
    end

    -- =========================================================================
    -- CreateTab
    -- =========================================================================
    function W:CreateTab(tabname)
        local tab = {
            button = Instance.new("TextButton", W.tab_container),
            frame  = Instance.new("Frame", W.content),
        }

        tab.button.Size                  = UDim2.fromOffset(0, 28)
        tab.button.AutomaticSize         = Enum.AutomaticSize.X
        tab.button.BackgroundTransparency = 1
        tab.button.Text                  = tabname
        tab.button.TextColor3            = library.theme.text_dark
        tab.button.Font                  = library.theme.font
        tab.button.TextSize              = 13
        tab.button.BorderSizePixel       = 0

        tab.frame.Size                  = UDim2.new(1, 0, 1, 0)
        tab.frame.BackgroundTransparency = 1
        tab.frame.Visible               = false

        -- ── Two columns ───────────────────────────────────────────────────────
        local function makeCol(xPos)
            local col          = Instance.new("ScrollingFrame", tab.frame)
            col.Size           = UDim2.new(0.5, -10, 1, 0)
            col.Position       = UDim2.new(xPos, xPos > 0 and 10 or 0, 0, 0)
            col.BackgroundTransparency = 1
            col.ScrollBarThickness = 2
            col.ScrollBarImageColor3 = library.theme.accent
            col.CanvasSize     = UDim2.new(0, 0, 0, 0)
            local layout       = Instance.new("UIListLayout", col)
            layout.Padding     = UDim.new(0, 10)
            layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
                col.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y + 8)
            end)
            return col
        end

        local left_col  = makeCol(0)
        local right_col = makeCol(0.5)

        -- Tab click
        tab.button.MouseButton1Click:Connect(function()
            for _, t in pairs(W.tabs) do
                t.frame.Visible       = false
                t.button.TextColor3   = library.theme.text_dark
            end
            tab.frame.Visible     = true
            tab.button.TextColor3 = library.theme.text

            -- Slide indicator
            tweensvc:Create(W.tab_indicator, TweenInfo.new(0.15, Enum.EasingStyle.Quad), {
                Position = UDim2.new(0, tab.button.AbsolutePosition.X - W.tab_container.AbsolutePosition.X, 1, 0),
                Size     = UDim2.fromOffset(tab.button.AbsoluteSize.X, 2),
            }):Play()
        end)

        if not W.active_tab then
            tab.frame.Visible     = true
            tab.button.TextColor3 = library.theme.text
            W.active_tab = tab
            -- init indicator after 1 frame
            task.defer(function()
                W.tab_indicator.Position = UDim2.new(0, tab.button.AbsolutePosition.X - W.tab_container.AbsolutePosition.X, 1, 0)
                W.tab_indicator.Size     = UDim2.fromOffset(tab.button.AbsoluteSize.X, 2)
            end)
        end

        -- =====================================================================
        -- CreateSector
        -- =====================================================================
        function tab:CreateSector(sname, side)
            W.sector_index = (W.sector_index % #library.theme.sector_colors) + 1
            local color    = library.theme.sector_colors[W.sector_index]
            local col      = side == "left" and left_col or right_col

            -- Outer wrapper
            local wrapper            = Instance.new("Frame", col)
            wrapper.Size             = UDim2.new(1, 0, 0, 0)
            wrapper.BackgroundColor3 = library.theme.surface
            wrapper.BorderSizePixel  = 0
            Instance.new("UICorner", wrapper).CornerRadius = UDim.new(0, 6)

            -- Colored left accent bar
            local accent_bar            = Instance.new("Frame", wrapper)
            accent_bar.Size             = UDim2.fromOffset(3, 0)
            accent_bar.Position         = UDim2.new(0, 0, 0, 0)
            accent_bar.BackgroundColor3 = color
            accent_bar.BorderSizePixel  = 0
            -- Corner only on left side via a nested frame trick
            local ab_corner = Instance.new("UICorner", accent_bar)
            ab_corner.CornerRadius = UDim.new(0, 4)

            -- Subtle top border glow
            local top_stroke = Instance.new("UIStroke", wrapper)
            top_stroke.Color       = color
            top_stroke.Thickness   = 1
            top_stroke.Transparency = 0.75

            -- Inner padding
            local inner            = Instance.new("Frame", wrapper)
            inner.Size             = UDim2.new(1, -12, 1, -8)
            inner.Position         = UDim2.new(0, 8, 0, 4)
            inner.BackgroundTransparency = 1

            local s_layout         = Instance.new("UIListLayout", inner)
            s_layout.Padding       = UDim.new(0, 8)

            local function updateHeight()
                local h = s_layout.AbsoluteContentSize.Y + 16
                wrapper.Size    = UDim2.new(1, 0, 0, h)
                accent_bar.Size = UDim2.fromOffset(3, h)
            end
            s_layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(updateHeight)

            -- Section title
            local s_title            = Instance.new("TextLabel", inner)
            s_title.Size             = UDim2.new(1, 0, 0, 18)
            s_title.Text             = sname
            s_title.TextColor3       = color
            s_title.Font             = library.theme.font
            s_title.TextSize         = 12
            s_title.TextXAlignment   = Enum.TextXAlignment.Left
            s_title.BackgroundTransparency = 1

            local sector = {}

            -- ── AddToggle ────────────────────────────────────────────────────
            function sector:AddToggle(text, default, callback)
                local toggle = {value = default}
                local btn    = Instance.new("TextButton", inner)
                btn.Size     = UDim2.new(1, 0, 0, 20)
                btn.BackgroundTransparency = 1
                btn.Text     = ""

                local box            = Instance.new("Frame", btn)
                box.Size             = UDim2.fromOffset(14, 14)
                box.Position         = UDim2.new(0, 0, 0.5, -7)
                box.BackgroundColor3 = library.theme.foreground
                box.BorderSizePixel  = 0
                Instance.new("UICorner", box).CornerRadius = UDim.new(0, 3)

                -- checkmark
                local check           = Instance.new("TextLabel", box)
                check.Size            = UDim2.new(1, 0, 1, 0)
                check.Text            = "✓"
                check.TextColor3      = library.theme.text
                check.Font            = library.theme.font
                check.TextSize        = 10
                check.BackgroundTransparency = 1
                check.Visible         = false

                local lbl             = Instance.new("TextLabel", btn)
                lbl.Size              = UDim2.new(1, -24, 1, 0)
                lbl.Position          = UDim2.new(0, 22, 0, 0)
                lbl.Text              = text
                lbl.TextColor3        = library.theme.text_dark
                lbl.Font              = library.theme.font
                lbl.TextSize          = 12
                lbl.TextXAlignment    = Enum.TextXAlignment.Left
                lbl.BackgroundTransparency = 1

                local function update()
                    tweensvc:Create(box, TweenInfo.new(0.12), {BackgroundColor3 = toggle.value and color or library.theme.foreground}):Play()
                    check.Visible     = toggle.value
                    lbl.TextColor3    = toggle.value and library.theme.text or library.theme.text_dark
                    callback(toggle.value)
                end

                btn.MouseButton1Click:Connect(function()
                    toggle.value = not toggle.value
                    update()
                end)
                update()
                return toggle
            end

            -- ── AddSlider ────────────────────────────────────────────────────
            function sector:AddSlider(text, min, default, max, step, callback)
                local s_cont            = Instance.new("Frame", inner)
                s_cont.Size             = UDim2.new(1, 0, 0, 36)
                s_cont.BackgroundTransparency = 1

                local lbl               = Instance.new("TextLabel", s_cont)
                lbl.Size                = UDim2.new(0.6, 0, 0, 16)
                lbl.Text                = text
                lbl.TextColor3          = library.theme.text_dark
                lbl.Font                = library.theme.font
                lbl.TextSize            = 12
                lbl.TextXAlignment      = Enum.TextXAlignment.Left
                lbl.BackgroundTransparency = 1

                local val_lbl           = Instance.new("TextLabel", s_cont)
                val_lbl.Size            = UDim2.new(0.4, 0, 0, 16)
                val_lbl.Position        = UDim2.new(0.6, 0, 0, 0)
                val_lbl.Text            = tostring(default)
                val_lbl.TextColor3      = color
                val_lbl.Font            = library.theme.font
                val_lbl.TextSize        = 12
                val_lbl.TextXAlignment  = Enum.TextXAlignment.Right
                val_lbl.BackgroundTransparency = 1

                local track             = Instance.new("Frame", s_cont)
                track.Size              = UDim2.new(1, 0, 0, 4)
                track.Position          = UDim2.new(0, 0, 0, 23)
                track.BackgroundColor3  = library.theme.foreground
                track.BorderSizePixel   = 0
                Instance.new("UICorner", track).CornerRadius = UDim.new(0, 2)

                local fill              = Instance.new("Frame", track)
                fill.Size               = UDim2.new((default - min) / (max - min), 0, 1, 0)
                fill.BackgroundColor3   = color
                fill.BorderSizePixel    = 0
                Instance.new("UICorner", fill).CornerRadius = UDim.new(0, 2)

                -- Thumb dot
                local thumb             = Instance.new("Frame", track)
                thumb.Size              = UDim2.fromOffset(10, 10)
                thumb.AnchorPoint       = Vector2.new(0.5, 0.5)
                thumb.Position          = UDim2.new((default - min) / (max - min), 0, 0.5, 0)
                thumb.BackgroundColor3  = color
                thumb.BorderSizePixel   = 0
                Instance.new("UICorner", thumb).CornerRadius = UDim.new(1, 0)

                local function setValue(percent)
                    local v = min + math.round((max - min) * percent / (step or 1)) * (step or 1)
                    v = math.clamp(v, min, max)
                    local p = (v - min) / (max - min)
                    fill.Size          = UDim2.new(p, 0, 1, 0)
                    thumb.Position     = UDim2.new(p, 0, 0.5, 0)
                    val_lbl.Text       = tostring(v)
                    callback(v)
                end

                track.InputBegan:Connect(function(inp)
                    if inp.UserInputType == Enum.UserInputType.MouseButton1 then
                        W.is_interacting = true
                        local conn
                        conn = uis.InputChanged:Connect(function(i)
                            if i.UserInputType == Enum.UserInputType.MouseMovement then
                                local p = math.clamp((uis:GetMouseLocation().X - track.AbsolutePosition.X) / track.AbsoluteSize.X, 0, 1)
                                setValue(p)
                            end
                        end)
                        uis.InputEnded:Connect(function(i)
                            if i.UserInputType == Enum.UserInputType.MouseButton1 then
                                W.is_interacting = false
                                conn:Disconnect()
                            end
                        end)
                    end
                end)
            end

            function sector:AddDropdown(text, options, default, callback)
                local d_cont            = Instance.new("Frame", inner)
                d_cont.Size             = UDim2.new(1, 0, 0, 26)

                d_cont.ClipsDescendants = false

                local drop              = Instance.new("TextButton", d_cont)

                drop.BackgroundColor3   = library.theme.foreground

                drop.TextColor3         = library.theme.text_dark
                drop.Font               = library.theme.font
                drop.TextSize           = 12
                drop.TextXAlignment     = Enum.TextXAlignment.Left
                drop.BorderSizePixel    = 0
                Instance.new("UICorner", drop).CornerRadius = UDim.new(0, 4)

                -- Arrow
                local arrow             = Instance.new("TextLabel", drop)
                arrow.Size              = UDim2.fromOffset(16, 16)
                arrow.Position          = UDim2.new(1, -20, 0.5, -8)
                arrow.Text              = "▾"
                arrow.TextColor3        = color
                arrow.Font              = library.theme.font
                arrow.TextSize          = 11
                arrow.BackgroundTransparency = 1

                local list              = Instance.new("Frame", d_cont)

                list.Position           = UDim2.new(0, 0, 1, 3)
                list.BackgroundColor3   = library.theme.foreground
                list.Visible            = false
                list.ZIndex             = 20
                list.BorderSizePixel    = 0
                list.ClipsDescendants   = true
                Instance.new("UICorner", list).CornerRadius = UDim.new(0, 4)
                Instance.new("UIListLayout", list)

                drop.MouseButton1Click:Connect(function()
                    list.Visible = not list.Visible
                    tweensvc:Create(list, TweenInfo.new(0.12), {
                        Size = list.Visible and UDim2.new(1, 0, 0, #options * 22) or UDim2.new(1, 0, 0, 0)
                    }):Play()
                end)

                for _, opt in ipairs(options) do
                    local ob = Instance.new("TextButton", list)
                    ob.Size              = UDim2.new(1, 0, 0, 22)
                    ob.BackgroundTransparency = 1
                    ob.Text              = "  " .. opt
                    ob.TextColor3        = library.theme.text_dark
                    ob.Font              = library.theme.font
                    ob.TextSize          = 12
                    ob.TextXAlignment    = Enum.TextXAlignment.Left
                    ob.ZIndex            = 21

                    ob.MouseEnter:Connect(function()
                        tweensvc:Create(ob, TweenInfo.new(0.08), {TextColor3 = library.theme.text}):Play()
                    end)
                    ob.MouseLeave:Connect(function()
                        tweensvc:Create(ob, TweenInfo.new(0.08), {TextColor3 = library.theme.text_dark}):Play()
                    end)

                    ob.MouseButton1Click:Connect(function()
                        drop.Text   = "  " .. text .. " : " .. opt
                        list.Visible = false
                        tweensvc:Create(list, TweenInfo.new(0.1), {Size = UDim2.new(1, 0, 0, 0)}):Play()
                        callback(opt)
                    end)
                end
            end

            -- ── AddKeybind ───────────────────────────────────────────────────
            function sector:AddKeybind(text, default, callback)
                local k_cont            = Instance.new("Frame", inner)
                k_cont.Size             = UDim2.new(1, 0, 0, 22)
                k_cont.BackgroundTransparency = 1

                local lbl               = Instance.new("TextLabel", k_cont)
                lbl.Size                = UDim2.new(0.58, 0, 1, 0)
                lbl.Text                = text
                lbl.TextColor3          = library.theme.text_dark
                lbl.Font                = library.theme.font
                lbl.TextSize            = 12
                lbl.TextXAlignment      = Enum.TextXAlignment.Left
                lbl.BackgroundTransparency = 1

                local kbtn              = Instance.new("TextButton", k_cont)
                kbtn.Size               = UDim2.new(0.38, 0, 1, 0)
                kbtn.Position           = UDim2.new(0.62, 0, 0, 0)
                kbtn.BackgroundColor3   = library.theme.foreground
                kbtn.Text               = default.Name
                kbtn.TextColor3         = color
                kbtn.Font               = library.theme.font
                kbtn.TextSize           = 11
                kbtn.BorderSizePixel    = 0
                Instance.new("UICorner", kbtn).CornerRadius = UDim.new(0, 3)

                local listening = false
                kbtn.MouseButton1Click:Connect(function()
                    listening   = true
                    kbtn.Text   = "..."
                    kbtn.TextColor3 = library.theme.text_dark
                end)
                uis.InputBegan:Connect(function(inp)
                    if listening and inp.UserInputType == Enum.UserInputType.Keyboard then
                        listening   = false
                        kbtn.Text   = inp.KeyCode.Name
                        kbtn.TextColor3 = color
                        callback(inp.KeyCode)
                    end
                end)
            end

            task.defer(updateHeight)
            return sector
        end

        table.insert(W.tabs, tab)
        return tab
    end

    -- =========================================================================
    -- CreatePlayerTab
    -- =========================================================================
    function W:CreatePlayerTab()
        local p_tab     = self:CreateTab("Players")
        local p_list_s  = p_tab:CreateSector("Player List", "left")
        local p_act_s   = p_tab:CreateSector("Actions", "right")

        local selected_player = nil
        local p_buttons       = {}

        -- We'll add player buttons directly to left column
        local left_col = p_list_s  -- forward reference via sector

        local function refresh()
            for _, b in ipairs(p_buttons) do b:Destroy() end
            p_buttons = {}
            -- Find left column frame parent
            local container = left_col
            for _, p in ipairs(players:GetPlayers()) do
                if p ~= players.LocalPlayer then
                    container:AddButton("  " .. p.Name, function()
                        selected_player = p
                    end)
                end
            end
        end

        p_act_s:AddButton("Teleport to Player", function()
            if selected_player and selected_player.Character then
                players.LocalPlayer.Character:MoveTo(selected_player.Character.HumanoidRootPart.Position)
            end
        end)
        p_act_s:AddButton("Spectate Player", function()
            if selected_player and selected_player.Character then
                workspace.CurrentCamera.CameraSubject = selected_player.Character.Humanoid
            end
        end)
        p_act_s:AddButton("Stop Spectate", function()
            workspace.CurrentCamera.CameraSubject = players.LocalPlayer.Character and players.LocalPlayer.Character.Humanoid
        end)

        refresh()
        players.PlayerAdded:Connect(refresh)
        players.PlayerRemoving:Connect(refresh)
    end

    -- =========================================================================
    -- CreateOptionsTab
    -- =========================================================================
    function W:CreateOptionsTab()
        local o_tab    = self:CreateTab("Options")
        local ui_sec   = o_tab:CreateSector("Interface", "left")
        local cfg_sec  = o_tab:CreateSector("Configuration", "right")

        -- ── UI Size selector ─────────────────────────────────────────────────
        ui_sec:AddDropdown("UI Size", {"Tiny (0.75x)", "Small (0.85x)", "Medium (1x)", "Large (1.15x)", "XLarge (1.3x)"}, "Medium (1x)", function(opt)
            local map = {
                ["Tiny (0.75x)"]   = 0.75,
                ["Small (0.85x)"]  = 0.85,
                ["Medium (1x)"]    = 1.0,
                ["Large (1.15x)"]  = 1.15,
                ["XLarge (1.3x)"]  = 1.30,
            }
            W:SetScale(map[opt] or 1.0)
        end)

        -- ── Accent color presets ──────────────────────────────────────────────
        ui_sec:AddButton("Theme: Violet (default)", function()
            library.theme.accent = Color3.fromRGB(100, 80, 255)
        end)
        ui_sec:AddButton("Theme: Rose", function()
            library.theme.accent = Color3.fromRGB(255, 60, 120)
        end)
        ui_sec:AddButton("Theme: Cyan", function()
            library.theme.accent = Color3.fromRGB(60, 200, 255)
        end)

        cfg_sec:AddKeybind("Toggle UI Key", W.hidebutton, function(key)
            W.hidebutton = key
        end)
        cfg_sec:AddLabel("Build: Feb 20 2026", library.theme.text_dark)
    end

    return W
end

-- ─── Notify ───────────────────────────────────────────────────────────────────

function library:Notify(title, text, duration)
    local notif_gui = Instance.new("ScreenGui", coregui)
    notif_gui.Name  = "EclipseNotif"

    local frame             = Instance.new("Frame", notif_gui)
    frame.Size              = UDim2.fromOffset(220, 50)
    frame.Position          = UDim2.new(1, -234, 1, -70)
    frame.BackgroundColor3  = library.theme.background
    frame.BorderSizePixel   = 0
    Instance.new("UICorner", frame).CornerRadius = UDim.new(0, 6)

    local stroke = Instance.new("UIStroke", frame)
    stroke.Color       = library.theme.accent
    stroke.Thickness   = 1
    stroke.Transparency = 0.4

    local t_lbl             = Instance.new("TextLabel", frame)
    t_lbl.Size              = UDim2.new(1, -10, 0, 18)
    t_lbl.Position          = UDim2.new(0, 8, 0, 6)
    t_lbl.Text              = title
    t_lbl.TextColor3        = library.theme.accent
    t_lbl.Font              = library.theme.font
    t_lbl.TextSize          = 12
    t_lbl.TextXAlignment    = Enum.TextXAlignment.Left
    t_lbl.BackgroundTransparency = 1

    local b_lbl             = Instance.new("TextLabel", frame)
    b_lbl.Size              = UDim2.new(1, -10, 0, 16)
    b_lbl.Position          = UDim2.new(0, 8, 0, 26)
    b_lbl.Text              = text
    b_lbl.TextColor3        = library.theme.text_dark
    b_lbl.Font              = library.theme.font
    b_lbl.TextSize          = 11
    b_lbl.TextXAlignment    = Enum.TextXAlignment.Left
    b_lbl.BackgroundTransparency = 1

    -- slide in
    frame.Position = UDim2.new(1, 10, 1, -70)
    tweensvc:Create(frame, TweenInfo.new(0.25, Enum.EasingStyle.Back), {
        Position = UDim2.new(1, -234, 1, -70)
    }):Play()

    task.delay(duration or 5, function()
        tweensvc:Create(frame, TweenInfo.new(0.2), {
            Position = UDim2.new(1, 10, 1, -70)
        }):Play()
        task.delay(0.25, function() notif_gui:Destroy() end)
    end)
end

return library
