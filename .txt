--[[
    MatchaLibrary_Ultimate.lua (V2 FIXED)
    Design: Matcha (Exact Photo Style - Ultimate Version)
    Structure: EchoLabs (CreateWindow, CreateTab, CreateSector, etc.)
    Features: Watermark (FPS/MS), Player List (TP/Spect), Options (Themes/Keybind)

local library = {
    flags = {},
    items = {},
    theme = {
        background = Color3.fromRGB(10, 10, 12),
        foreground = Color3.fromRGB(20, 20, 25),
        accent = Color3.fromRGB(255, 0, 150), -- Rose Matcha
        text = Color3.fromRGB(255, 255, 255),
        text_dark = Color3.fromRGB(150, 150, 160),
        font = Enum.Font.GothamBold,
    }
}

local uis = game:GetService("UserInputService")
local tweenservice = game:GetService("TweenService")
local coregui = game:GetService("CoreGui")
local players = game:GetService("Players")
local runservice = game:GetService("RunService")
local stats = game:GetService("Stats")

-- Watermark (FPS/MS)
function library:CreateWatermark()
    local watermark = Instance.new("ScreenGui", coregui)
    watermark.Name = "MatchaWatermark"

    local frame = Instance.new("Frame", watermark)
    frame.Size = UDim2.new(0, 180, 0, 30)
    frame.Position = UDim2.new(0, 10, 0, 10)
    frame.BackgroundColor3 = library.theme.background
    frame.BorderSizePixel = 0
    local corner = Instance.new("UICorner", frame)
    corner.CornerRadius = UDim.new(0, 4)

    local label = Instance.new("TextLabel", frame)
    label.Size = UDim2.new(1, -10, 1, 0)
    label.Position = UDim2.new(0, 10, 0, 0)
    label.BackgroundTransparency = 1
    label.Text = "Matcha | 0 FPS | 0 MS"
    label.TextColor3 = library.theme.text
    label.Font = library.theme.font
    label.TextSize = 13
    label.TextXAlignment = Enum.TextXAlignment.Left

    runservice.RenderStepped:Connect(function()
        local fps = math.floor(1 / runservice.RenderStepped:Wait())
        local ping = math.floor(stats.Network.ServerStatsItem["Data Ping"]:GetValue())
        label.Text = "Matcha | " .. fps .. " FPS | " .. ping .. " MS"
    end)
end

function library:CreateWindow(name, size, hidebutton)
    local window = {
        tabs = {},
        active_tab = nil,
        hidebutton = hidebutton or Enum.KeyCode.V,
        visible = true
    }

    -- GUI Principal
    window.main = Instance.new("ScreenGui", coregui)
    window.main.Name = name

    window.frame = Instance.new("Frame", window.main)
    window.frame.Size = UDim2.fromOffset(650, 500)
    window.frame.Position = UDim2.new(0.6, 0, 0.5, -250)
    window.frame.BackgroundColor3 = library.theme.background
    window.frame.BorderSizePixel = 0
    window.frame.Active = true
    window.frame.Draggable = true

    local corner = Instance.new("UICorner", window.frame)
    corner.CornerRadius = UDim.new(0, 6)

    -- Header
    local header = Instance.new("Frame", window.frame)
    header.Size = UDim2.new(1, -40, 0, 40)
    header.Position = UDim2.new(0, 20, 0, 10)
    header.BackgroundTransparency = 1

    local title = Instance.new("TextLabel", header)
    title.Size = UDim2.new(1, 0, 1, 0)
    title.Text = "Matcha  Comfort  beta"
    title.TextColor3 = library.theme.accent
    title.Font = library.theme.font
    title.TextSize = 16
    title.TextXAlignment = Enum.TextXAlignment.Left
    title.BackgroundTransparency = 1

    -- Tabs Container
    window.tab_container = Instance.new("Frame", window.frame)
    window.tab_container.Size = UDim2.new(1, -40, 0, 30)
    window.tab_container.Position = UDim2.new(0, 20, 0, 50)
    window.tab_container.BackgroundTransparency = 1

    local tab_layout = Instance.new("UIListLayout", window.tab_container)
    tab_layout.FillDirection = Enum.FillDirection.Horizontal
    tab_layout.Padding = UDim.new(0, 20)

    -- Content Frame
    window.content = Instance.new("Frame", window.frame)
    window.content.Size = UDim2.new(1, -40, 1, -100)
    window.content.Position = UDim2.new(0, 20, 0, 90)
    window.content.BackgroundTransparency = 1

    -- Toggle UI Visibility
    uis.InputBegan:Connect(function(input, gpe)
        if not gpe and input.KeyCode == window.hidebutton then
            window.visible = not window.visible
            window.frame.Visible = window.visible
        end
    end)

    function window:CreateTab(name)
        local tab = {
            button = Instance.new("TextButton", window.tab_container),
            frame = Instance.new("Frame", window.content)
        }

        tab.button.Size = UDim2.new(0, 60, 1, 0)
        tab.button.BackgroundTransparency = 1
        tab.button.Text = name
        tab.button.TextColor3 = library.theme.text_dark
        tab.button.Font = library.theme.font
        tab.button.TextSize = 14
        tab.button.BorderSizePixel = 0

        tab.frame.Size = UDim2.new(1, 0, 1, 0)
        tab.frame.BackgroundTransparency = 1
        tab.frame.Visible = false

        -- Columns
        local left_col = Instance.new("ScrollingFrame", tab.frame)
        left_col.Size = UDim2.new(0.5, -10, 1, 0)
        left_col.BackgroundTransparency = 1
        left_col.ScrollBarThickness = 0
        left_col.CanvasSize = UDim2.new(0, 0, 0, 0)
        local l_layout = Instance.new("UIListLayout", left_col)
        l_layout.Padding = UDim.new(0, 20)
        l_layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
            left_col.CanvasSize = UDim2.new(0, 0, 0, l_layout.AbsoluteContentSize.Y)
        end)

        local right_col = Instance.new("ScrollingFrame", tab.frame)
        right_col.Size = UDim2.new(0.5, -10, 1, 0)
        right_col.Position = UDim2.new(0.5, 10, 0, 0)
        right_col.BackgroundTransparency = 1
        right_col.ScrollBarThickness = 0
        right_col.CanvasSize = UDim2.new(0, 0, 0, 0)
        local r_layout = Instance.new("UIListLayout", right_col)
        r_layout.Padding = UDim.new(0, 20)
        r_layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
            right_col.CanvasSize = UDim2.new(0, 0, 0, r_layout.AbsoluteContentSize.Y)
        end)

        tab.button.MouseButton1Click:Connect(function()
            for _, t in pairs(window.tabs) do
                t.frame.Visible = false
                t.button.TextColor3 = library.theme.text_dark
            end
            tab.frame.Visible = true
            tab.button.TextColor3 = library.theme.text
        end)

        if not window.active_tab then
            tab.frame.Visible = true
            tab.button.TextColor3 = library.theme.text
            window.active_tab = tab
        end

        function tab:CreateSector(name, side)
            local sector = {
                container = side == "left" and left_col or right_col
            }
            
            local s_frame = Instance.new("Frame", sector.container)
            s_frame.Size = UDim2.new(1, 0, 0, 0)
            s_frame.BackgroundTransparency = 1
            
            local s_layout = Instance.new("UIListLayout", s_frame)
            s_layout.Padding = UDim.new(0, 10)
            s_layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
                s_frame.Size = UDim2.new(1, 0, 0, s_layout.AbsoluteContentSize.Y)
            end)

            local s_title = Instance.new("TextLabel", s_frame)
            s_title.Size = UDim2.new(1, 0, 0, 20)
            s_title.Text = name
            s_title.TextColor3 = library.theme.accent
            s_title.Font = library.theme.font
            s_title.TextSize = 14
            s_title.TextXAlignment = Enum.TextXAlignment.Left
            s_title.BackgroundTransparency = 1

            function sector:AddToggle(text, default, callback)
                local toggle = {value = default}
                local btn = Instance.new("TextButton", s_frame)
                btn.Size = UDim2.new(1, 0, 0, 20)
                btn.BackgroundTransparency = 1
                btn.Text = ""

                local box = Instance.new("Frame", btn)
                box.Size = UDim2.fromOffset(16, 16)
                box.Position = UDim2.new(0, 0, 0.5, -8)
                box.BackgroundColor3 = library.theme.foreground
                box.BorderSizePixel = 0
                local b_corner = Instance.new("UICorner", box)
                b_corner.CornerRadius = UDim.new(0, 4)

                local label = Instance.new("TextLabel", btn)
                label.Size = UDim2.new(1, -25, 1, 0)
                label.Position = UDim2.new(0, 25, 0, 0)
                label.Text = text
                label.TextColor3 = library.theme.text_dark
                label.Font = library.theme.font
                label.TextSize = 13
                label.TextXAlignment = Enum.TextXAlignment.Left
                label.BackgroundTransparency = 1

                local function update()
                    box.BackgroundColor3 = toggle.value and library.theme.accent or library.theme.foreground
                    label.TextColor3 = toggle.value and library.theme.text or library.theme.text_dark
                    callback(toggle.value)
                end

                btn.MouseButton1Click:Connect(function()
                    toggle.value = not toggle.value
                    update()
                end)
                update()
                return toggle
            end

            function sector:AddSlider(text, min, default, max, step, callback)
                local container = Instance.new("Frame", s_frame)
                container.Size = UDim2.new(1, 0, 0, 35)
                container.BackgroundTransparency = 1

                local label = Instance.new("TextLabel", container)
                label.Size = UDim2.new(0.5, 0, 0, 15)
                label.Text = text
                label.TextColor3 = library.theme.text_dark
                label.Font = library.theme.font
                label.TextSize = 13
                label.TextXAlignment = Enum.TextXAlignment.Left
                label.BackgroundTransparency = 1

                local val = Instance.new("TextLabel", container)
                val.Size = UDim2.new(0.5, 0, 0, 15)
                val.Position = UDim2.new(0.5, 0, 0, 0)
                val.Text = tostring(default)
                val.TextColor3 = library.theme.text
                val.Font = library.theme.font
                val.TextSize = 13
                val.TextXAlignment = Enum.TextXAlignment.Right
                val.BackgroundTransparency = 1

                local bar = Instance.new("Frame", container)
                bar.Size = UDim2.new(1, 0, 0, 4)
                bar.Position = UDim2.new(0, 0, 0, 22)
                bar.BackgroundColor3 = library.theme.foreground
                bar.BorderSizePixel = 0
                local bar_corner = Instance.new("UICorner", bar)
                bar_corner.CornerRadius = UDim.new(0, 2)

                local fill = Instance.new("Frame", bar)
                fill.Size = UDim2.new((default-min)/(max-min), 0, 1, 0)
                fill.BackgroundColor3 = library.theme.accent
                fill.BorderSizePixel = 0
                local fill_corner = Instance.new("UICorner", fill)
                fill_corner.CornerRadius = UDim.new(0, 2)

                bar.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then
                        local connection
                        connection = uis.InputChanged:Connect(function(input)
                            if input.UserInputType == Enum.UserInputType.MouseMovement then
                                local mouse_pos = uis:GetMouseLocation().X
                                local bar_pos = bar.AbsolutePosition.X
                                local bar_size = bar.AbsoluteSize.X
                                local percent = math.clamp((mouse_pos - bar_pos) / bar_size, 0, 1)
                                local value = math.floor(min + (max - min) * percent)
                                fill.Size = UDim2.new(percent, 0, 1, 0)
                                val.Text = tostring(value)
                                callback(value)
                            end
                        end)
                        uis.InputEnded:Connect(function(input)
                            if input.UserInputType == Enum.UserInputType.MouseButton1 then
                                connection:Disconnect()
                            end
                        end)
                    end
                end)
            end

            function sector:AddDropdown(text, options, default, callback)
                local drop = Instance.new("TextButton", s_frame)
                drop.Size = UDim2.new(1, 0, 0, 25)
                drop.BackgroundColor3 = library.theme.foreground
                drop.Text = "  " .. text .. " : " .. (default or "None")
                drop.TextColor3 = library.theme.text_dark
                drop.Font = library.theme.font
                drop.TextSize = 13
                drop.TextXAlignment = Enum.TextXAlignment.Left
                local drop_corner = Instance.new("UICorner", drop)
                drop_corner.CornerRadius = UDim.new(0, 4)
            end

            function sector:AddButton(text, callback)
                local btn = Instance.new("TextButton", s_frame)
                btn.Size = UDim2.new(1, 0, 0, 25)
                btn.BackgroundColor3 = library.theme.foreground
                btn.Text = text
                btn.TextColor3 = library.theme.text
                btn.Font = library.theme.font
                btn.TextSize = 13
                btn.MouseButton1Click:Connect(callback)
                local btn_corner = Instance.new("UICorner", btn)
                btn_corner.CornerRadius = UDim.new(0, 4)
            end

            function sector:AddLabel(text, color)
                local label = Instance.new("TextLabel", s_frame)
                label.Size = UDim2.new(1, 0, 0, 20)
                label.Text = text
                label.TextColor3 = color or library.theme.text_dark
                label.Font = library.theme.font
                label.TextSize = 13
                label.TextXAlignment = Enum.TextXAlignment.Left
                label.BackgroundTransparency = 1
            end

            return sector
        end

        table.insert(window.tabs, tab)
        return tab
    end

    function window:CreatePlayerTab()
        local p_tab = self:CreateTab("Players")
        local p_list = p_tab:CreateSector("Player List", "left")
        local p_actions = p_tab:CreateSector("Actions", "right")

        local selected_player = nil

        local function refresh_list()
            for _, child in pairs(p_list.container:GetChildren()) do
                if child:IsA("TextButton") then child:Destroy() end
            end
            for _, p in pairs(players:GetPlayers()) do
                if p ~= players.LocalPlayer then
                    local p_btn = Instance.new("TextButton", p_list.container)
                    p_btn.Size = UDim2.new(1, 0, 0, 25)
                    p_btn.BackgroundColor3 = library.theme.foreground
                    p_btn.Text = "  " .. p.Name
                    p_btn.TextColor3 = library.theme.text_dark
                    p_btn.Font = library.theme.font
                    p_btn.TextSize = 13
                    p_btn.TextXAlignment = Enum.TextXAlignment.Left
                    local p_corner = Instance.new("UICorner", p_btn)
                    p_corner.CornerRadius = UDim.new(0, 4)
                    
                    p_btn.MouseButton1Click:Connect(function()
                        selected_player = p
                        p_actions:AddLabel("Selected: " .. p.Name, library.theme.accent)
                    end)
                end
            end
        end

        p_actions:AddButton("Teleport", function()
            if selected_player and selected_player.Character then
                players.LocalPlayer.Character:MoveTo(selected_player.Character.HumanoidRootPart.Position)
            end
        end)

        p_actions:AddButton("Spectate", function()
            if selected_player and selected_player.Character then
                workspace.CurrentCamera.CameraSubject = selected_player.Character.Humanoid
            end
        end)

        p_actions:AddButton("Stop Spectate", function()
            workspace.CurrentCamera.CameraSubject = players.LocalPlayer.Character.Humanoid
        end)

        refresh_list()
        players.PlayerAdded:Connect(refresh_list)
        players.PlayerRemoving:Connect(refresh_list)
    end

    function window:CreateOptionsTab()
        local o_tab = self:CreateTab("Options")
        local themes = o_tab:CreateSector("Themes", "left")
        local config = o_tab:CreateSector("Config", "right")

        themes:AddButton("Reset Accent (Matcha Rose)", function()
            library.theme.accent = Color3.fromRGB(255, 0, 150)
        end)

        config:AddLabel("UI Keybind: V")
    end

    return window
end

function library:Notify(title, text, duration)
    print("[" .. title .. "] " .. text)
end

return library
