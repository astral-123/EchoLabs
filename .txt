local CF = "phantom_configs"
local AL = "phantom_autoload.txt"

local function safe(fn,...) local ok,r=pcall(fn,...) return ok and r or nil end

local function ensureFolder()
    if isfolder and not isfolder(CF) and makefolder then makefolder(CF) end
end
local function listConfigs()
    ensureFolder()
    local files = safe(listfiles, CF) or {}
    local names = {}
    for _,p in ipairs(files) do
        local n = p:match("[/\\]?([^/\\]+)%.json$")
        if n then table.insert(names, n) end
    end
    return names
end
local function saveConfigFile(name, data)
    ensureFolder()
    if writefile then
        writefile(CF.."/"..name..".json", http:JSONEncode(data))
        return true
    end
    return false
end
local function loadConfigFile(name)
    local raw = safe(readfile, CF.."/"..name..".json")
    if raw then return safe(http.JSONDecode, http, raw) end
    return nil
end
local function deleteConfigFile(name)
    local p = CF.."/"..name..".json"
    if isfile and isfile(p) and delfile then delfile(p) end
end
local function getAutoLoad()
    if isfile and isfile(AL) and readfile then
        local v = safe(readfile, AL)
        return (v and v ~= "") and v or nil
    end
end
local function setAutoLoad(name)
    if writefile then writefile(AL, name or "") end
end

local function corner(p,r)
    local c=Instance.new("UICorner",p);c.CornerRadius=UDim.new(0,r or 4);return c
end
local function mks(p,color,thick,transp)
    local s=Instance.new("UIStroke",p)
    s.Color=color or library.theme.surface
    s.Thickness=thick or 1
    s.Transparency=transp or 0
    return s
end

local function mkColorPicker(anchor, cur_col, on_change)
    for _,c in ipairs(coregui:GetChildren()) do
        if c.Name=="PhCP" then c:Destroy() end
    end
    local sg=Instance.new("ScreenGui",coregui);sg.Name="PhCP"
    sg.ZIndexBehavior=Enum.ZIndexBehavior.Sibling

    local pop=Instance.new("Frame",sg)
    pop.Size=UDim2.fromOffset(210,215)
    pop.BackgroundColor3=library.theme.bg
    pop.BorderSizePixel=0;pop.ZIndex=100
    corner(pop,6);mks(pop,library.theme.accent,1,0.5)
    local abs=anchor.AbsolutePosition
    pop.Position=UDim2.fromOffset(abs.X, abs.Y+anchor.AbsoluteSize.Y+4)

    local h,s,v=Color3.toHSV(cur_col)

    local prev=Instance.new("Frame",pop)
    prev.Size=UDim2.new(1,-16,0,22);prev.Position=UDim2.new(0,8,0,8)
    prev.BackgroundColor3=cur_col;prev.BorderSizePixel=0;corner(prev,4)

    -- Hue bar
    local ht=Instance.new("Frame",pop)
    ht.Size=UDim2.new(1,-16,0,10);ht.Position=UDim2.new(0,8,0,38)
    ht.BackgroundColor3=Color3.new(1,1,1);ht.BorderSizePixel=0;corner(ht,3)
    local hg=Instance.new("UIGradient",ht)
    hg.Color=ColorSequence.new({
        ColorSequenceKeypoint.new(0,Color3.fromHSV(0,1,1)),
        ColorSequenceKeypoint.new(0.17,Color3.fromHSV(0.17,1,1)),
        ColorSequenceKeypoint.new(0.33,Color3.fromHSV(0.33,1,1)),
        ColorSequenceKeypoint.new(0.5,Color3.fromHSV(0.5,1,1)),
        ColorSequenceKeypoint.new(0.67,Color3.fromHSV(0.67,1,1)),
        ColorSequenceKeypoint.new(0.83,Color3.fromHSV(0.83,1,1)),
        ColorSequenceKeypoint.new(1,Color3.fromHSV(1,1,1)),
    })
    local hth=Instance.new("Frame",ht)
    hth.Size=UDim2.fromOffset(4,14);hth.AnchorPoint=Vector2.new(0.5,0.5)
    hth.Position=UDim2.new(h,0,0.5,0);hth.BackgroundColor3=Color3.new(1,1,1)
    hth.BorderSizePixel=0;corner(hth,2)

    -- SV box
    local svf=Instance.new("Frame",pop)
    svf.Size=UDim2.new(1,-16,0,105);svf.Position=UDim2.new(0,8,0,56)
    svf.BorderSizePixel=0;corner(svf,4)
    local svb=Instance.new("Frame",svf)
    svb.Size=UDim2.new(1,0,1,0);svb.BackgroundColor3=Color3.fromHSV(h,1,1)
    svb.BorderSizePixel=0;corner(svb,4)
    local sg2=Instance.new("UIGradient",svb)
    sg2.Color=ColorSequence.new(Color3.new(1,1,1),Color3.fromHSV(h,1,1))
    local svdk=Instance.new("Frame",svb)
    svdk.Size=UDim2.new(1,0,1,0);svdk.BackgroundColor3=Color3.new(0,0,0)
    svdk.BorderSizePixel=0;corner(svdk,4)
    local dg=Instance.new("UIGradient",svdk)
    dg.Color=ColorSequence.new(Color3.new(0,0,0),Color3.new(0,0,0))
    dg.Transparency=NumberSequence.new({NumberSequenceKeypoint.new(0,1),NumberSequenceKeypoint.new(1,0)})
    dg.Rotation=90
    local svth=Instance.new("Frame",svf)
    svth.Size=UDim2.fromOffset(8,8);svth.AnchorPoint=Vector2.new(0.5,0.5)
    svth.Position=UDim2.new(s,0,1-v,0);svth.BackgroundColor3=Color3.new(1,1,1)
    svth.BorderSizePixel=0;svth.ZIndex=5;corner(svth,10);mks(svth,Color3.new(0,0,0),1)

    local function upd()
        local col=Color3.fromHSV(h,s,v)
        prev.BackgroundColor3=col
        svb.BackgroundColor3=Color3.fromHSV(h,1,1)
        sg2.Color=ColorSequence.new(Color3.new(1,1,1),Color3.fromHSV(h,1,1))
        on_change(col)
    end

    local hd,svd=false,false
    ht.InputBegan:Connect(function(i) if i.UserInputType==Enum.UserInputType.MouseButton1 then hd=true end end)
    svf.InputBegan:Connect(function(i) if i.UserInputType==Enum.UserInputType.MouseButton1 then svd=true end end)
    uis.InputChanged:Connect(function(i)
        if i.UserInputType~=Enum.UserInputType.MouseMovement then return end
        local mx=uis:GetMouseLocation()
        if hd then
            h=math.clamp((mx.X-ht.AbsolutePosition.X)/ht.AbsoluteSize.X,0,1)
            hth.Position=UDim2.new(h,0,0.5,0);upd()
        end
        if svd then
            s=math.clamp((mx.X-svf.AbsolutePosition.X)/svf.AbsoluteSize.X,0,1)
            v=1-math.clamp((mx.Y-svf.AbsolutePosition.Y)/svf.AbsoluteSize.Y,0,1)
            svth.Position=UDim2.new(s,0,1-v,0);upd()
        end
    end)
    uis.InputEnded:Connect(function(i)
        if i.UserInputType==Enum.UserInputType.MouseButton1 then hd=false;svd=false end
    end)

    local cb=Instance.new("TextButton",pop)
    cb.Size=UDim2.new(1,-16,0,18);cb.Position=UDim2.new(0,8,1,-24)
    cb.BackgroundColor3=library.theme.surface;cb.Text="Fermer"
    cb.TextColor3=library.theme.dim;cb.Font=library.theme.font;cb.TextSize=11
    cb.BorderSizePixel=0;corner(cb,3)
    cb.MouseButton1Click:Connect(function() sg:Destroy() end)

    local conn
    conn=uis.InputBegan:Connect(function(i)
        if i.UserInputType==Enum.UserInputType.MouseButton1 then
            local mx=uis:GetMouseLocation()
            local px,py=pop.AbsolutePosition.X,pop.AbsolutePosition.Y
            if mx.X<px or mx.X>px+pop.AbsoluteSize.X or mx.Y<py or mx.Y>py+pop.AbsoluteSize.Y then
                sg:Destroy();conn:Disconnect()
            end
        end
    end)
end

            fill.BackgroundColor3=library.theme.accent;fill.BorderSizePixel=0;corner(fill,2)
            local function setV(p)
                local val=math.clamp(math.floor(min+(max-min)*p/(step or 1))*(step or 1),min,max)
                fill.Size=UDim2.new(p,0,1,0);vl.Text=tostring(val);state.value=val
                if flagName then library.flags[flagName]=val end
                if callback then callback(val) end
            end
            state.setValue=function(val)
                local p=math.clamp((val-min)/(max-min),0,1);setV(p)
            end

                    W.is_interacting=true

            if flagName then
                library.flags[flagName]=state.value
                library._elements[flagName]={type="dropdown",ref=state,cb=callback}

            lbl_.TextXAlignment=Enum.TextXAlignment.Left

            local drop=Instance.new("TextButton",cont)
            drop.Size=UDim2.new(1,0,0,20);drop.Position=UDim2.new(0,0,0,14)
            drop.BackgroundColor3=library.theme.surface
            drop.Text="  "..(default or options[1] or "")
            drop.TextColor3=library.theme.text;drop.Font=library.theme.font_reg
            drop.TextSize=12;drop.TextXAlignment=Enum.TextXAlignment.Left
            drop.BorderSizePixel=0;corner(drop,3)

            local arrow=Instance.new("TextLabel",drop)
            arrow.Size=UDim2.fromOffset(18,20);arrow.Position=UDim2.new(1,-20,0,0)
            arrow.BackgroundTransparency=1;arrow.Text="▼";arrow.TextColor3=library.theme.accent
            arrow.Font=library.theme.font;arrow.TextSize=9;arrow.TextXAlignment=Enum.TextXAlignment.Center

            -- ⚡ List is parented to W.sg (top-level ScreenGui) so it is NEVER clipped
            local list=Instance.new("Frame", W.sg)
            list.Size=UDim2.fromOffset(0,0)   -- starts collapsed
            list.BackgroundColor3=library.theme.surface
            list.Visible=false
            list.ZIndex=200                   -- always on top of everything
            list.BorderSizePixel=0
            list.ClipsDescendants=true
            corner(list,3)
            mks(list, library.theme.accent, 1, 0.7)

                open=false;list.Visible=false
                tw:Create(list,TweenInfo.new(0.08),{Size=UDim2.fromOffset(drop.AbsoluteSize.X,0)}):Play()
                tw:Create(arrow,TweenInfo.new(0.1),{Rotation=0}):Play()
            end

            local function openList()
                -- Reposition list under the drop button in absolute screen coords
                local abs=drop.AbsolutePosition

            end

            deleteConfigFile(n);cur.name=nil
            refreshDropdown()
            library:Notify("Config","Supprimée : "..n,4)
        end)

        cfg_s:AddButton("Set Auto Load",function()
            if not cur.name then library:Notify("Config","Aucune config sélectionnée",4);return end
            setAutoLoad(cur.name);autoload_cfg=cur.name
            library:Notify("Config","Auto Load → "..cur.name,4)
        end)

        cfg_s:AddButton("Clear Auto Load",function()
            setAutoLoad("");autoload_cfg=nil
            library:Notify("Config","Auto Load supprimé",4)
        end)

        -- Auto-load on start
        if autoload_cfg then
            task.defer(function() task.wait(0.5);applyConfig(autoload_cfg) end)
        end
    end

    return W
end

-- Notify
function library:Notify(title, text, duration)
    local sg=Instance.new("ScreenGui",coregui);sg.Name="PhantomNotif"
    local f=Instance.new("Frame",sg)
    f.Size=UDim2.fromOffset(210,46);f.Position=UDim2.new(1,10,1,-62)
    f.BackgroundColor3=library.theme.bg;f.BorderSizePixel=0
    corner(f,5);mks(f,library.theme.accent,1,0.4)
    local t_=Instance.new("TextLabel",f)
    t_.Size=UDim2.new(1,-8,0,18);t_.Position=UDim2.new(0,8,0,5);t_.BackgroundTransparency=1
    t_.Text=title;t_.TextColor3=library.theme.accent;t_.Font=library.theme.font;t_.TextSize=12
    t_.TextXAlignment=Enum.TextXAlignment.Left
    local b_=Instance.new("TextLabel",f)
    b_.Size=UDim2.new(1,-8,0,16);b_.Position=UDim2.new(0,8,0,24);b_.BackgroundTransparency=1
    b_.Text=text;b_.TextColor3=library.theme.dim;b_.Font=library.theme.font_reg;b_.TextSize=11
    b_.TextXAlignment=Enum.TextXAlignment.Left
    tw:Create(f,TweenInfo.new(0.2,Enum.EasingStyle.Back),{Position=UDim2.new(1,-222,1,-62)}):Play()
    task.delay(duration or 5,function()
        tw:Create(f,TweenInfo.new(0.15),{Position=UDim2.new(1,10,1,-62)}):Play()
        task.delay(0.2,function() sg:Destroy() end)
    end)
end

return library
