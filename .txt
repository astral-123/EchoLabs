-- ─────────────────────────────────────────────────────────────

function library:CreateWatermark()
    local sg = Instance.new("ScreenGui", coregui)
    sg.Name = "MatchaWatermark"

    local f = Instance.new("Frame", sg)
    f.Size            = UDim2.fromOffset(195, 24)
    f.Position        = UDim2.new(0, 12, 0, 12)
    f.BackgroundColor3 = library.theme.bg
    f.BorderSizePixel = 0
    Instance.new("UICorner", f).CornerRadius = UDim.new(0, 4)
    local s = Instance.new("UIStroke", f)
    s.Color = library.theme.surface ; s.Thickness = 1

    local lbl = Instance.new("TextLabel", f)
    lbl.Size = UDim2.new(1, -8, 1, 0)
    lbl.Position = UDim2.new(0, 8, 0, 0)
    lbl.BackgroundTransparency = 1
    lbl.Text = "matcha | 0 FPS | 0 MS"
    lbl.TextColor3 = library.theme.text
    lbl.Font = library.theme.font
    lbl.TextSize = 12
    lbl.TextXAlignment = Enum.TextXAlignment.Left

    -- pink dot
    local dot = Instance.new("Frame", f)
    dot.Size = UDim2.fromOffset(5,5)
    dot.Position = UDim2.new(0,6,0.5,-2)
    dot.BackgroundColor3 = library.theme.accent
    dot.BorderSizePixel = 0
    Instance.new("UICorner",dot).CornerRadius = UDim.new(1,0)
    lbl.Position = UDim2.new(0,16,0,0)

    run.RenderStepped:Connect(function()
        local fps = math.floor(1/(run.RenderStepped:Wait()))
        local ok,ping = pcall(function() return math.floor(stats.Network.ServerStatsItem["Data Ping"]:GetValue()) end)
        lbl.Text = "matcha | "..fps.." FPS | "..(ok and ping or 0).." MS"
    end)
end

-- ─────────────────────────────────────────────────────────────
-- WINDOW
-- ─────────────────────────────────────────────────────────────
function library:CreateWindow(title, hidekey)
    hidekey = hidekey or Enum.KeyCode.Insert

    local W = { tabs={}, active_tab=nil, visible=true, dragging=false, is_interacting=false }

    W.sg = Instance.new("ScreenGui", coregui)
    W.sg.Name = "MatchaUI"

    -- Main frame (500 × 540, positionné à droite comme sur la photo)
    W.frame = Instance.new("Frame", W.sg)
    W.frame.Size = UDim2.fromOffset(500, 540)
    W.frame.Position = UDim2.new(1, -515, 0.5, -270)
    W.frame.BackgroundColor3 = library.theme.bg
    W.frame.BorderSizePixel = 0
    W.frame.Active = true
    Instance.new("UICorner", W.frame).CornerRadius = UDim.new(0,6)
    local main_stroke = Instance.new("UIStroke", W.frame)
    main_stroke.Color = library.theme.surface
    main_stroke.Thickness = 1

    -- ── Drag ──────────────────────────────────────────────────
    local dInput, dStart, dPos
    W.frame.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 and not W.is_interacting then
            W.dragging = true ; dStart = i.Position ; dPos = W.frame.Position
            i.Changed:Connect(function() if i.UserInputState == Enum.UserInputState.End then W.dragging=false end end)
        end
    end)
    W.frame.InputChanged:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseMovement then dInput=i end
    end)
    uis.InputChanged:Connect(function(i)
        if i==dInput and W.dragging then
            local d=i.Position-dStart
            W.frame.Position=UDim2.new(dPos.X.Scale,dPos.X.Offset+d.X,dPos.Y.Scale,dPos.Y.Offset+d.Y)
        end
    end)

    -- ── Header ────────────────────────────────────────────────
    local header = Instance.new("Frame", W.frame)
    header.Size = UDim2.new(1,0,0,32)
    header.BackgroundTransparency = 1

    local title_lbl = Instance.new("TextLabel", header)
    title_lbl.Size = UDim2.new(0.7,0,1,0)
    title_lbl.Position = UDim2.new(0,14,0,0)
    title_lbl.BackgroundTransparency = 1
    title_lbl.Text = "Matcha  |  Comfort  |  beta"
    title_lbl.TextColor3 = library.theme.accent
    title_lbl.Font = library.theme.font
    title_lbl.TextSize = 13
    title_lbl.TextXAlignment = Enum.TextXAlignment.Left

    local user_lbl = Instance.new("TextLabel", header)
    user_lbl.Size = UDim2.new(0.3,-12,1,0)
    user_lbl.Position = UDim2.new(0.7,0,0,0)
    user_lbl.BackgroundTransparency = 1
    user_lbl.Text = "zixlr"
    user_lbl.TextColor3 = library.theme.dim
    user_lbl.Font = library.theme.font_reg
    user_lbl.TextSize = 12
    user_lbl.TextXAlignment = Enum.TextXAlignment.Right

    -- Thin separator under header
    local sep1 = Instance.new("Frame", W.frame)
    sep1.Size = UDim2.new(1,0,0,1)
    sep1.Position = UDim2.new(0,0,0,32)
    sep1.BackgroundColor3 = library.theme.surface
    sep1.BorderSizePixel = 0

    -- ── Main Tab Bar ──────────────────────────────────────────
    local tabbar = Instance.new("Frame", W.frame)
    tabbar.Size = UDim2.new(1,0,0,28)
    tabbar.Position = UDim2.new(0,0,0,33)
    tabbar.BackgroundTransparency = 1

    local tabbar_layout = Instance.new("UIListLayout", tabbar)
    tabbar_layout.FillDirection = Enum.FillDirection.Horizontal
    tabbar_layout.Padding = UDim.new(0,0)

    -- Active tab underline indicator
    W.indicator = Instance.new("Frame", tabbar)
    W.indicator.Size = UDim2.fromOffset(60,2)
    W.indicator.Position = UDim2.new(0,0,1,-2)
    W.indicator.BackgroundColor3 = library.theme.accent
    W.indicator.BorderSizePixel = 0
    W.indicator.ZIndex = 10

    local sep2 = Instance.new("Frame", W.frame)
    sep2.Size = UDim2.new(1,0,0,1)
    sep2.Position = UDim2.new(0,0,0,61)
    sep2.BackgroundColor3 = library.theme.surface
    sep2.BorderSizePixel = 0

    -- ── Content area ──────────────────────────────────────────
    W.content = Instance.new("Frame", W.frame)
    W.content.Size = UDim2.new(1,0,1,-63)
    W.content.Position = UDim2.new(0,0,0,63)
    W.content.BackgroundTransparency = 1

    -- hide/show toggle
    uis.InputBegan:Connect(function(i,gpe)
        if not gpe and i.KeyCode==hidekey then
            W.visible = not W.visible
            W.frame.Visible = W.visible
        end
    end)

    -- ================================================================
    -- CreateTab  (main tab: Combat, Visuals, etc.)
    -- ================================================================
    function W:CreateTab(name)
        local tab = {
            name=name,
            left_subtabs={},
            right_subtabs={},
            active_left=nil,
            active_right=nil,
        }

        -- Tab button
        tab.btn = Instance.new("TextButton", tabbar)
        tab.btn.Size = UDim2.fromOffset(string.len(name)*8+20, 28)
        tab.btn.BackgroundTransparency = 1
        tab.btn.Text = name
        tab.btn.TextColor3 = library.theme.dim
        tab.btn.Font = library.theme.font
        tab.btn.TextSize = 13
        tab.btn.BorderSizePixel = 0

        -- Tab content frame
        tab.frame = Instance.new("Frame", W.content)
        tab.frame.Size = UDim2.new(1,0,1,0)
        tab.frame.BackgroundTransparency = 1
        tab.frame.Visible = false

        -- Two columns separator line
        local col_sep = Instance.new("Frame", tab.frame)
        col_sep.Size = UDim2.fromOffset(1, 99999)
        col_sep.Position = UDim2.new(0.5,0,0,0)
        col_sep.BackgroundColor3 = library.theme.surface
        col_sep.BorderSizePixel = 0

        -- LEFT sub-tab bar
        tab.left_bar = Instance.new("Frame", tab.frame)
        tab.left_bar.Size = UDim2.new(0.5,-1,0,26)
        tab.left_bar.BackgroundTransparency = 1
        local ll = Instance.new("UIListLayout", tab.left_bar)
        ll.FillDirection = Enum.FillDirection.Horizontal
        ll.Padding = UDim.new(0,18)
        local llp = Instance.new("UIPadding", tab.left_bar)
        llp.PaddingLeft = UDim.new(0,12)

        -- LEFT sub-tab content
        tab.left_content = Instance.new("Frame", tab.frame)
        tab.left_content.Size = UDim2.new(0.5,-1,1,-27)
        tab.left_content.Position = UDim2.new(0,0,0,27)
        tab.left_content.BackgroundTransparency = 1

        -- LEFT sub-tab indicator
        tab.left_ind = Instance.new("Frame", tab.left_bar)
        tab.left_ind.Size = UDim2.fromOffset(40,1)
        tab.left_ind.Position = UDim2.new(0,12,1,-1)
        tab.left_ind.BackgroundColor3 = library.theme.accent
        tab.left_ind.BorderSizePixel = 0
        tab.left_ind.ZIndex = 5

        -- RIGHT sub-tab bar
        tab.right_bar = Instance.new("Frame", tab.frame)
        tab.right_bar.Size = UDim2.new(0.5,-1,0,26)
        tab.right_bar.Position = UDim2.new(0.5,1,0,0)
        tab.right_bar.BackgroundTransparency = 1
        local rl = Instance.new("UIListLayout", tab.right_bar)
        rl.FillDirection = Enum.FillDirection.Horizontal
        rl.Padding = UDim.new(0,18)
        local rlp = Instance.new("UIPadding", tab.right_bar)
        rlp.PaddingLeft = UDim.new(0,12)

        -- RIGHT sub-tab content
        tab.right_content = Instance.new("Frame", tab.frame)
        tab.right_content.Size = UDim2.new(0.5,-1,1,-27)
        tab.right_content.Position = UDim2.new(0.5,1,0,27)
        tab.right_content.BackgroundTransparency = 1

        -- RIGHT sub-tab indicator
        tab.right_ind = Instance.new("Frame", tab.right_bar)
        tab.right_ind.Size = UDim2.fromOffset(40,1)
        tab.right_ind.Position = UDim2.new(0,12,1,-1)
        tab.right_ind.BackgroundColor3 = library.theme.accent
        tab.right_ind.BorderSizePixel = 0
        tab.right_ind.ZIndex = 5

        -- Separator below sub-tab bars
        local sep3 = Instance.new("Frame", tab.frame)
        sep3.Size = UDim2.new(1,0,0,1)
        sep3.Position = UDim2.new(0,0,0,26)
        sep3.BackgroundColor3 = library.theme.surface
        sep3.BorderSizePixel = 0

        -- Main tab click
        tab.btn.MouseButton1Click:Connect(function()
            for _,t in ipairs(W.tabs) do
                t.frame.Visible = false
                t.btn.TextColor3 = library.theme.dim
            end
            tab.frame.Visible = true
            tab.btn.TextColor3 = library.theme.text
            tween:Create(W.indicator, TweenInfo.new(0.12), {
                Position = UDim2.new(0, tab.btn.AbsolutePosition.X - tabbar.AbsolutePosition.X, 1, -2),
                Size     = UDim2.fromOffset(tab.btn.AbsoluteSize.X, 2)
            }):Play()
        end)

        -- First tab auto-open
        if not W.active_tab then
            W.active_tab = tab
            tab.frame.Visible = true
            tab.btn.TextColor3 = library.theme.text
            task.defer(function()
                W.indicator.Position = UDim2.new(0, tab.btn.AbsolutePosition.X - tabbar.AbsolutePosition.X, 1, -2)
                W.indicator.Size = UDim2.fromOffset(tab.btn.AbsoluteSize.X, 2)
            end)
        end

        -- ============================================================
        -- Helper : build element into a scroll frame
        -- ============================================================
        local function makeScroll(parent)
            local scroll = Instance.new("ScrollingFrame", parent)
            scroll.Size = UDim2.new(1,0,1,0)
            scroll.BackgroundTransparency = 1
            scroll.ScrollBarThickness = 2
            scroll.ScrollBarImageColor3 = library.theme.accent
            scroll.CanvasSize = UDim2.new(0,0,0,0)
            local layout = Instance.new("UIListLayout", scroll)
            layout.Padding = UDim.new(0,0)
            local pad = Instance.new("UIPadding", scroll)
            pad.PaddingLeft = UDim.new(0,10)
            pad.PaddingRight = UDim.new(0,8)
            pad.PaddingTop = UDim.new(0,6)
            layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
                scroll.CanvasSize = UDim2.new(0,0,0,layout.AbsoluteContentSize.Y+10)
            end)
            return scroll
        end

        -- ============================================================
        -- BuildElements  — attaches AddToggle/AddSlider etc to a scroll
        -- ============================================================
        local function buildElements(scroll)
            local api = {}

            -- ── AddToggle ───────────────────────────────────────────
            -- text, default, callback, keybind_default (optional)
            function api:AddToggle(text, default, callback, kb_default)
                local row = Instance.new("TextButton", scroll)
                row.Size = UDim2.new(1,0,0,22)
                row.BackgroundTransparency = 1
                row.Text = ""
                row.BorderSizePixel = 0

                local box = Instance.new("Frame", row)
                box.Size = UDim2.fromOffset(13,13)
                box.Position = UDim2.new(0,0,0.5,-6)
                box.BackgroundColor3 = library.theme.surface
                box.BorderSizePixel = 0
                Instance.new("UICorner",box).CornerRadius = UDim.new(0,3)
                local box_stroke = Instance.new("UIStroke",box)
                box_stroke.Color = library.theme.dim
                box_stroke.Thickness = 1

                local lbl = Instance.new("TextLabel", row)
                lbl.Size = UDim2.new(0.65,0,1,0)
                lbl.Position = UDim2.new(0,20,0,0)
                lbl.BackgroundTransparency = 1
                lbl.Text = text
                lbl.TextColor3 = library.theme.dim
                lbl.Font = library.theme.font_reg
                lbl.TextSize = 13
                lbl.TextXAlignment = Enum.TextXAlignment.Left

                -- Optional keybind badge on right
                local kb_btn
                if kb_default then
                    kb_btn = Instance.new("TextButton", row)
                    kb_btn.Size = UDim2.fromOffset(60,16)
                    kb_btn.Position = UDim2.new(1,-62,0.5,-8)
                    kb_btn.BackgroundColor3 = library.theme.surface
                    kb_btn.Text = kb_default.Name
                    kb_btn.TextColor3 = library.theme.dim
                    kb_btn.Font = library.theme.font
                    kb_btn.TextSize = 11
                    kb_btn.BorderSizePixel = 0
                    Instance.new("UICorner",kb_btn).CornerRadius=UDim.new(0,3)

                    local listening = false
                    kb_btn.MouseButton1Click:Connect(function()
                        listening = true ; kb_btn.Text = "..."
                    end)
                    uis.InputBegan:Connect(function(i)
                        if listening and i.UserInputType==Enum.UserInputType.Keyboard then
                            listening=false ; kb_btn.Text=i.KeyCode.Name
                        end
                    end)
                end

                local val = {value=default}
                local function update()
                    box.BackgroundColor3 = val.value and library.theme.accent or library.theme.surface
                    box_stroke.Color     = val.value and library.theme.accent or library.theme.dim
                    lbl.TextColor3       = val.value and library.theme.text or library.theme.dim
                    if callback then callback(val.value) end
                end
                row.MouseButton1Click:Connect(function()
                    val.value = not val.value ; update()
                end)
                update()
                return val
            end

            -- ── AddSlider ───────────────────────────────────────────
            function api:AddSlider(text, min, default, max, step, callback)
                local container = Instance.new("Frame", scroll)
                container.Size = UDim2.new(1,0,0,38)
                container.BackgroundTransparency = 1

                local lbl = Instance.new("TextLabel", container)
                lbl.Size = UDim2.new(0.65,0,0,18)
                lbl.BackgroundTransparency = 1
                lbl.Text = text
                lbl.TextColor3 = library.theme.dim
                lbl.Font = library.theme.font_reg
                lbl.TextSize = 13
                lbl.TextXAlignment = Enum.TextXAlignment.Left

                local val_lbl = Instance.new("TextLabel", container)
                val_lbl.Size = UDim2.new(0.35,-2,0,18)
                val_lbl.Position = UDim2.new(0.65,0,0,0)
                val_lbl.BackgroundTransparency = 1
                val_lbl.Text = tostring(default)
                val_lbl.TextColor3 = library.theme.text
                val_lbl.Font = library.theme.font
                val_lbl.TextSize = 13
                val_lbl.TextXAlignment = Enum.TextXAlignment.Right

                local track = Instance.new("Frame", container)
                track.Size = UDim2.new(1,0,0,3)
                track.Position = UDim2.new(0,0,0,24)
                track.BackgroundColor3 = library.theme.surface
                track.BorderSizePixel = 0
                Instance.new("UICorner",track).CornerRadius = UDim.new(0,2)

                local fill = Instance.new("Frame", track)
                fill.Size = UDim2.new((default-min)/(max-min),0,1,0)
                fill.BackgroundColor3 = library.theme.accent
                fill.BorderSizePixel = 0
                Instance.new("UICorner",fill).CornerRadius = UDim.new(0,2)

                track.InputBegan:Connect(function(i)
                    if i.UserInputType==Enum.UserInputType.MouseButton1 then
                        W.is_interacting=true
                        local c c=uis.InputChanged:Connect(function(i2)
                            if i2.UserInputType==Enum.UserInputType.MouseMovement then
                                local p=math.clamp((uis:GetMouseLocation().X-track.AbsolutePosition.X)/track.AbsoluteSize.X,0,1)
                                local v=math.floor(min+(max-min)*p)
                                fill.Size=UDim2.new(p,0,1,0)
                                val_lbl.Text=tostring(v)
                                if callback then callback(v) end
                            end
                        end)
                        uis.InputEnded:Connect(function(i2)
                            if i2.UserInputType==Enum.UserInputType.MouseButton1 then
                                W.is_interacting=false ; c:Disconnect()
                            end
                        end)
                    end
                end)
            end

                cont.Size = UDim2.new(1,0,0,30)

                    ob.TextColor3 = library.theme.dim
                    ob.Font = library.theme.font_reg
                    ob.TextSize = 12
                    ob.TextXAlignment = Enum.TextXAlignment.Left
                    ob.ZIndex = 51
                    ob.MouseEnter:Connect(function() ob.TextColor3=library.theme.text end)
                    ob.MouseLeave:Connect(function() ob.TextColor3=library.theme.dim end)
                    ob.MouseButton1Click:Connect(function()
                        drop.Text = "  "..opt
                        list.Visible=false
                        tween:Create(list,TweenInfo.new(0.08),{Size=UDim2.new(1,0,0,0)}):Play()
                        if callback then callback(opt) end
                    end)
                end
            end

        -- ============================================================
        -- CreateSector  — crée un sous-onglet (Aimbot, Silent Aim…)
        -- side = "left" | "right"
        -- ============================================================
        function tab:CreateSector(name, side)
            local bar     = side=="left" and tab.left_bar     or tab.right_bar
            local content = side=="left" and tab.left_content or tab.right_content
            local ind     = side=="left" and tab.left_ind     or tab.right_ind
            local list    = side=="left" and tab.left_subtabs  or tab.right_subtabs

            local sector = {}

            -- Sub-tab button
            sector.btn = Instance.new("TextButton", bar)
            sector.btn.Size = UDim2.fromOffset(0,26)
            sector.btn.AutomaticSize = Enum.AutomaticSize.X
            sector.btn.BackgroundTransparency = 1
            sector.btn.Text = name
            sector.btn.TextColor3 = library.theme.dim
            sector.btn.Font = library.theme.font
            sector.btn.TextSize = 12
            sector.btn.BorderSizePixel = 0

            -- Sub-tab scroll content
            sector.scroll = makeScroll(content)
            sector.scroll.Visible = false

            -- Attach element builders
            local api = buildElements(sector.scroll)
            for k,v in pairs(api) do sector[k] = v end

            sector.btn.MouseButton1Click:Connect(function()
                for _,s in ipairs(list) do
                    s.scroll.Visible = false
                    s.btn.TextColor3 = library.theme.dim
                end
                sector.scroll.Visible = true
                sector.btn.TextColor3 = library.theme.text
                tween:Create(ind,TweenInfo.new(0.12),{
                    Position = UDim2.new(0, sector.btn.AbsolutePosition.X - bar.AbsolutePosition.X, 1,-1),
                    Size     = UDim2.fromOffset(sector.btn.AbsoluteSize.X,1)
                }):Play()
            end)

            -- Auto-activate first sector
            if #list==0 then
                sector.scroll.Visible = true
                sector.btn.TextColor3 = library.theme.text
                task.defer(function()
                    ind.Position = UDim2.new(0, sector.btn.AbsolutePosition.X - bar.AbsolutePosition.X, 1,-1)
                    ind.Size     = UDim2.fromOffset(sector.btn.AbsoluteSize.X,1)
                end)
            end

            table.insert(list, sector)
            return sector
        end

        table.insert(W.tabs, tab)
        return tab
    end

    -- ================================================================
    -- Players Tab (exact photo layout)
    -- ================================================================
    function W:CreatePlayerTab()
        local pt = self:CreateTab("Players")
        local list_sec = pt:CreateSector("Players", "left")
        local act_sec  = pt:CreateSector("Actions", "right")

        local selected = nil

        local function refresh()
            -- clear existing player buttons
            for _,c in ipairs(list_sec.scroll:GetChildren()) do
                if c:IsA("TextButton") and c.Text ~= "" then c:Destroy() end
            end
            for _,p in ipairs(players:GetPlayers()) do
                if p ~= players.LocalPlayer then
                    local pb = Instance.new("TextButton", list_sec.scroll)
                    pb.Size = UDim2.new(1,0,0,26)
                    pb.BackgroundColor3 = library.theme.surface
                    pb.Text = "  "..p.Name
                    pb.TextColor3 = library.theme.dim
                    pb.Font = library.theme.font_reg
                    pb.TextSize = 12
                    pb.TextXAlignment = Enum.TextXAlignment.Left
                    pb.BorderSizePixel = 0
                    Instance.new("UICorner",pb).CornerRadius = UDim.new(0,4)
                    pb.MouseButton1Click:Connect(function()
                        selected = p
                        for _,c2 in ipairs(list_sec.scroll:GetChildren()) do
                            if c2:IsA("TextButton") then
                                c2.BackgroundColor3 = library.theme.surface
                                c2.TextColor3 = library.theme.dim
                            end
                        end
                        pb.BackgroundColor3 = library.theme.accent:Lerp(Color3.new(0,0,0),0.5)
                        pb.TextColor3 = library.theme.text
                    end)
                end
            end
        end

        act_sec:AddButton("Teleport to Player", function()
            if selected and selected.Character then
                players.LocalPlayer.Character:MoveTo(selected.Character.HumanoidRootPart.Position)
            end
        end)
        act_sec:AddButton("Spectate", function()
            if selected and selected.Character then
                workspace.CurrentCamera.CameraSubject = selected.Character.Humanoid
            end
        end)
        act_sec:AddButton("Stop Spectate", function()
            workspace.CurrentCamera.CameraSubject = players.LocalPlayer.Character.Humanoid
        end)

        refresh()
        players.PlayerAdded:Connect(refresh)
        players.PlayerRemoving:Connect(refresh)
    end

    -- ================================================================
    -- Options Tab
    -- ================================================================
    function W:CreateOptionsTab()
        local ot = self:CreateTab("Options")
        local ui_s = ot:CreateSector("UI", "left")
        local kb_s = ot:CreateSector("Keybinds", "right")

        ui_s:AddDropdown("UI Size", {"Tiny","Small","Normal","Large","XLarge"}, "Normal", function(opt)
            local map = {Tiny=0.75,Small=0.85,Normal=1,Large=1.15,XLarge=1.3}
            local f = map[opt] or 1
            W.frame.Size = UDim2.fromOffset(math.floor(500*f), math.floor(540*f))
        end)

        ui_s:AddButton("Accent: Rose",  function() library.theme.accent=Color3.fromRGB(255,0,140) end)
        ui_s:AddButton("Accent: Violet",function() library.theme.accent=Color3.fromRGB(120,60,255) end)
        ui_s:AddButton("Accent: Cyan",  function() library.theme.accent=Color3.fromRGB(0,200,255) end)

        kb_s:AddKeybind("Toggle UI", hidekey, function(k) hidekey=k end)
    end

    return W
end

function library:Notify(title, text, duration)
    local sg = Instance.new("ScreenGui",coregui) ; sg.Name="MatchaNotif"
    local f = Instance.new("Frame",sg)
    f.Size = UDim2.fromOffset(210,46)
    f.Position = UDim2.new(1,10,1,-62)
    f.BackgroundColor3 = library.theme.bg
    f.BorderSizePixel = 0
    Instance.new("UICorner",f).CornerRadius = UDim.new(0,5)
    local s=Instance.new("UIStroke",f);s.Color=library.theme.accent;s.Thickness=1;s.Transparency=0.4

    local t=Instance.new("TextLabel",f)
    t.Size=UDim2.new(1,-8,0,18);t.Position=UDim2.new(0,8,0,5)
    t.BackgroundTransparency=1;t.Text=title;t.TextColor3=library.theme.accent
    t.Font=library.theme.font;t.TextSize=12;t.TextXAlignment=Enum.TextXAlignment.Left

    local b=Instance.new("TextLabel",f)
    b.Size=UDim2.new(1,-8,0,16);b.Position=UDim2.new(0,8,0,24)
    b.BackgroundTransparency=1;b.Text=text;b.TextColor3=library.theme.dim
    b.Font=library.theme.font_reg;b.TextSize=11;b.TextXAlignment=Enum.TextXAlignment.Left

    tween:Create(f,TweenInfo.new(0.2,Enum.EasingStyle.Back),{Position=UDim2.new(1,-222,1,-62)}):Play()
    task.delay(duration or 5,function()
        tween:Create(f,TweenInfo.new(0.15),{Position=UDim2.new(1,10,1,-62)}):Play()
        task.delay(0.2,function() sg:Destroy() end)
    end)
end

return library
